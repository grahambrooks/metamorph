package com.amplify.loaner.activities.failure;

import com.amplify.loaner.R;
import com.google.common.collect.Lists;

import java.util.Collection;
import java.util.List;

public enum ResetFailure {
    DELETE_GCM_REGISTRATION_ID(R.string.delete_registration_id_failed),
    SHUTTING_DOWN_NETWORK_CONNECTION(R.string.network_shutdown_failure),
    DELETING_DATA(R.string.data_deletion_failure),
    REMOVING_ACCOUNTS(R.string.account_removal_failure),
    ADAM_DISASSOCIATION_FAILURE(R.string.failed_to_disassociate_with_adam);

    private int stringId;

    ResetFailure(int stringId) {
        this.stringId = stringId;
    }

    public int stringId() {
        return stringId;
    }

    public static Collection createFromName(List resetFailureKeys) {
        List resetFailures = Lists.newArrayList();
        for(String failure : resetFailureKeys){
            resetFailures.add(ResetFailure.valueOf(failure.toUpperCase())); //NOPMD
        }
        return resetFailures;
    }

    @Override
    public String toString() {
        return "ResetFailure{" +
                "stringId=" + stringId +
                '}';
    }
}
