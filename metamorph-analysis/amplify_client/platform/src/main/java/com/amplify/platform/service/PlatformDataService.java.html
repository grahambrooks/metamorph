package com.amplify.platform.service;

import android.content.Intent;
import android.os.IBinder;
import com.alpha.logging.LogUtils;
import com.amplify.platform.factory.TaskFactory;
import com.amplify.platform.task.EventRetrievalTask;
import com.google.inject.Inject;
import org.slf4j.Logger;
import roboguice.service.RoboService;

public class PlatformDataService extends RoboService {

    private static final Logger LOGGER = LogUtils.getLogger();
    public static final String AUTH_TOKEN = "AuthToken";
    @Inject
    private TaskFactory taskFactory;

    @Override
    public IBinder onBind(Intent intent) {

        String authToken = intent.getExtras().getString(AUTH_TOKEN);
        LOGGER.info("Platform Data Service Bound");
        EventRetrievalTask eventRetrievalTask = taskFactory.createEventRetrievalTask(authToken);
        eventRetrievalTask.execute();

        return null;
    }

}
