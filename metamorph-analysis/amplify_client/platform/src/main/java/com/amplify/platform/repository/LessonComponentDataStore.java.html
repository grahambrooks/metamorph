package com.amplify.platform.repository;

import android.content.Context;
import com.amplify.platform.domain.LinkLessonComponent;
import com.google.inject.Inject;
import com.google.inject.Singleton;

import static com.alpha.api.repository.sqlite.Table.LINK_LESSON_COMPONENTS;

@Singleton
public class LessonComponentDataStore {


    private final Context context;

    @Inject
    public LessonComponentDataStore(Context context) {
        this.context = context;
    }

    public void save(LinkLessonComponent linkLessonComponent) {
        context.getContentResolver().insert(LINK_LESSON_COMPONENTS.uri(), linkLessonComponent.asContentValues());
    }

    public void deleteAll() {
        context.getContentResolver().delete(LINK_LESSON_COMPONENTS.uri(), "1=1", null);
    }
}
