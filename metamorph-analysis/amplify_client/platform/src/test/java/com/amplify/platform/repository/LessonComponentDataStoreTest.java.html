package com.amplify.platform.repository;

import android.content.ContentResolver;
import android.content.ContentValues;
import android.content.Context;
import com.alpha.api.repository.sqlite.Table;
import com.amplify.platform.domain.LinkLessonComponent;
import com.amplify.runner.SundaeRobolectricTestRunner;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.Mockito.verify;
import static org.mockito.Mockito.when;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class LessonComponentDataStoreTest {

    private LessonComponentDataStore lessonComponentDataStore;

    @Mock
    private Context context;
    @Mock
    private ContentResolver contentResolver;
    @Mock
    private LinkLessonComponent linkLessonComponent;
    @Mock
    private ContentValues contentValues;

    @Before
    public void setUp() throws Exception {
        initMocks(this);

        lessonComponentDataStore = new LessonComponentDataStore(context);
    }

    @Test
    public void shouldInsertAnEventOnSave() {
        when(context.getContentResolver()).thenReturn(contentResolver);
        when(linkLessonComponent.asContentValues()).thenReturn(contentValues);

        lessonComponentDataStore.save(linkLessonComponent);

        verify(contentResolver).insert(Table.LINK_LESSON_COMPONENTS.uri(), contentValues);

    }


    @Test
    public void shouldDeleteAllOnDeleteAllCalled() {
        when(context.getContentResolver()).thenReturn(contentResolver);

        lessonComponentDataStore.deleteAll();

        verify(contentResolver).delete(Table.LINK_LESSON_COMPONENTS.uri(), "1=1", null);
    }
}
