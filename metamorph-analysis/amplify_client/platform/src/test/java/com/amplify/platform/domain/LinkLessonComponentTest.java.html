package com.amplify.platform.domain;

import android.content.ContentValues;
import com.alpha.api.repository.sqlite.TableColumn;
import com.amplify.runner.SundaeRobolectricTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.hamcrest.CoreMatchers.is;
import static org.junit.Assert.assertThat;

@RunWith(SundaeRobolectricTestRunner.class)
public class LinkLessonComponentTest {

    public static final String URL = "URL";
    public static final String TITLE = "A TITLE";
    public static final Long ID = 1L;
    public static final String IMAGE_URL = "IMAGE_URL";

    @Test
    public void shouldReturnContentValue() {

        LinkLessonComponent component = new LinkLessonComponent(ID, TITLE, URL, IMAGE_URL);

        ContentValues values = component.asContentValues();

        assertThat(values.getAsLong(TableColumn._id.name()), is(ID));
        assertThat(values.getAsString(TableColumn.title.name()), is(TITLE));
        assertThat(values.getAsString(TableColumn.url.name()), is(URL));
        assertThat(values.getAsString(TableColumn.image_url.name()), is(IMAGE_URL));
    }
}
