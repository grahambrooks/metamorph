package com.alpha.api.json;

import android.net.Uri;
import com.alpha.api.domains.LearningElement;
import com.alpha.api.domains.Response;
import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import com.google.gson.JsonElement;
import com.google.gson.JsonParser;

import java.lang.reflect.Type;
import java.util.Date;

public class AmplifyJsonHelper {
    private static final Gson GSON = new GsonBuilder()
            .registerTypeAdapter(Date.class, new CustomDateJsonAdapter())
            .registerTypeAdapter(LearningElement.class, new LearningElementJsonAdapter())
            .registerTypeAdapter(Response.class, new ResponseJsonAdapter())
            .registerTypeAdapter(Uri.class, new UriJsonAdapter())
            .create();

    private AmplifyJsonHelper() {
    }

    public static  T fromJson(String s, Class aClass) {
        return GSON.fromJson(s, aClass);
    }

    public static  T fromJson(String s, Type typeOfT) {
        return GSON.fromJson(s, typeOfT);
    }

    public static  T fromJson(JsonElement json, Class type) {
        return GSON.fromJson(json, type);
    }

    public static  T fromJson(String json, String memberName, Class classType) {
        return fromJson(new JsonParser().parse(json).getAsJsonObject().get(memberName), classType);
    }

    public static String toJson(Object o) {
        return GSON.toJson(o);
    }

    public static JsonElement toJsonTree(Object o, Type typeSrc) {
        return GSON.toJsonTree(o, typeSrc);
    }
}
