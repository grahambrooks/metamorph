package com.alpha.api.services;

import com.alpha.api.callbacks.ControlServerRegistrationCallback;

import java.util.Date;

/**
 * Interface/API to manage all retrieval for user related information
 */
public interface UserService {

    /**
     * Checks to see if a user has registered
     * @return TRUE if user has, FALSE otherwise
     */
    boolean hasUserRegistered();

    /**
     * Registers a user that is retrieved from services, i.e. an existing users. OR
     * Registers a user with the given username and role. Implementations should only accept names with numbers and letters or
     * letters only. Symbol/Number only names should be rejected.
     *
     * @param userToRegister user retrieved from backend or new instance with username supplied
     * @param callback The callback instance to process the result of the registration
     */
    void registerUser(User userToRegister, ControlServerRegistrationCallback callback);

    /**
     * Return the user that has been registered
     *
     * @return user thats been registered, NULL otherwise
     */
     T getUser();

    Section getCurrentUserSection();

    ClassSession session();

    boolean isTeacher();

    boolean isStudent();

    void saveAccountName(String accountName);

    String getAccountName();

    void syncAllStudentsForAllSections(User user);

    void refreshUser(User newUser);

    boolean isSessionOngoingForCurrentSection();

    boolean isSessionOngoingFor(Section section);

    String getCalledOnStudentName();

    void updateMostRecentPostTime(Date modifiedOn);

    Date getLastModifiedPostTime();
}
