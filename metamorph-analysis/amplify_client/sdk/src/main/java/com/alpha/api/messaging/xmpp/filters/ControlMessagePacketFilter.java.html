package com.alpha.api.messaging.xmpp.filters;

import com.google.common.base.Strings;
import org.jivesoftware.smack.packet.Message;

public class ControlMessagePacketFilter extends BasePacketFilter {
    @Override
    protected boolean acceptPacket(Message message) {
        String subject = message.getSubject();
        return !Strings.isNullOrEmpty(subject);
    }
}
