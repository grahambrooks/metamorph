package com.alpha.api.messaging.xmpp.events;

import com.alpha.api.messaging.ControlSettingsModel;
import com.alpha.api.services.ControlEvent;

public class CountdownTimerPredicate extends EventPredicates.EventPredicate {

    public CountdownTimerPredicate(ControlSettingsModel originalSettings, ControlSettingsModel newSettings) {
        super(originalSettings, newSettings);
    }

    @Override
    public ControlEvent event() {
        return ControlEvent.COUNTDOWN_TIMER;
    }

    @Override
    public boolean matches() {
        return isDifferent(originalSettings.getCountdownTimerSettings().isRunning(), newSettings.getCountdownTimerSettings().isRunning());
    }
}
