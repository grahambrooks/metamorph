package com.alpha.api.domains;

import android.content.Intent;
import android.graphics.Bitmap;
import android.net.Uri;
import android.provider.MediaStore;
import android.view.ViewGroup;
import com.google.gson.annotations.Expose;

import java.io.File;

import static android.media.ThumbnailUtils.createVideoThumbnail;

public class VideoLearningElement extends LearningElement {
    public static final String EXTRA_MEDIA_PATH = "extra_media_path";
    public static final String EXTRA_MEDIA_TITLE = "extra_media_title";
    @Expose
    private final String pathToVideo;

    public VideoLearningElement(String title, String description, String pathToVideo) {
        super(title, description, LearningElementType.VIDEO, null);
        this.pathToVideo = pathToVideo;
    }

    @Override

    public Bitmap getThumbnailBitmap() {
        return createVideoThumbnail(pathToVideo, MediaStore.Video.Thumbnails.MINI_KIND);
    }

    public String getPathToVideo() {
        return pathToVideo;
    }

    public Intent getVideoShareIntent() {
        Intent shareIntent = new Intent();
        shareIntent.putExtra(EXTRA_MEDIA_PATH, pathToVideo);
        shareIntent.putExtra(EXTRA_MEDIA_TITLE, getTitle());
        return shareIntent;
    }

    public Intent getVideoLaunchIntent() {
        Intent intent = new Intent(Intent.ACTION_VIEW);
        intent.setDataAndType(Uri.fromFile(new File(pathToVideo)), "video/*");
        intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        return intent;
    }
}
