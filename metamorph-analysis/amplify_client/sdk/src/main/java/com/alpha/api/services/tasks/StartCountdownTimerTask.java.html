package com.alpha.api.services.tasks;

import com.alpha.api.messaging.ControlSettingsModel;
import com.alpha.api.messaging.TeacherSession;
import com.alpha.api.services.User;
import com.alpha.api.services.UserService;
import com.google.inject.Inject;
import roboguice.util.SafeAsyncTask;

public class StartCountdownTimerTask extends SafeAsyncTask {

    private final UserService userService;
    private final TeacherSession teacherSession;

    @Inject
    public StartCountdownTimerTask(UserService userService, TeacherSession teacherSession) {
        this.userService = userService;
        this.teacherSession = teacherSession;
    }

    @Override
    public Boolean call() throws Exception {
        User user = userService.getUser();
        ControlSettingsModel currentControlSettingsModel = user.getCurrentControlSettingsModel();
        currentControlSettingsModel.getCountdownTimerSettings().convertToAbsoluteExpireTime();
        teacherSession.sendStudentControlSettings(currentControlSettingsModel);
        return true;
    }
}
