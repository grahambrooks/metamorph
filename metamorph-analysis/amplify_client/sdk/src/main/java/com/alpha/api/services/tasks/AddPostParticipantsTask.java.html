package com.alpha.api.services.tasks;

import com.alpha.api.authentication.ApiRequest;
import com.alpha.api.json.AmplifyJsonHelper;
import com.alpha.api.properties.Environment;
import com.alpha.api.services.SafeTimedAsyncTask;
import com.alpha.api.services.web.WebServices;
import com.alpha.logging.AnalyticsAgent;
import com.alpha.logging.LogUtils;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;
import org.slf4j.Logger;

import java.util.Collection;

public class AddPostParticipantsTask extends SafeTimedAsyncTask {
    private static final Logger LOG = LogUtils.getLogger();

    private final WebServices webServices;
    private final Environment environment;
    private final Long postId;
    private final Collection activeParticipants;

    @Inject
    public AddPostParticipantsTask(WebServices webServices,
                                   Environment environment,
                                   AnalyticsAgent analyticsAgent,
                                   @Assisted Long postId,
                                   @Assisted Collection activeParticipants) {

        super(analyticsAgent);
        this.webServices = webServices;
        this.environment = environment;
        this.postId = postId;
        this.activeParticipants = activeParticipants;
    }

    @Override
    public Void call() {
        final String uri = environment.getAddParticipantsToPostUri(postId);
        final String json = AmplifyJsonHelper.toJson(activeParticipants);
        webServices.put(ApiRequest.createPutRequest(uri, json));
        return null;
    }
