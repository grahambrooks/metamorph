package com.alpha.api.messaging;

import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

public class CountdownTimerSettings extends QuickActivitySettings {

    private Integer countdownTime;
    private Date absoluteTime;

    public Integer getCountdownTime() {
        return countdownTime;
    }

    public void setCountdownTime(Integer countdownTime) {
        this.countdownTime = countdownTime;
    }

    public void convertToAbsoluteExpireTime() {
        Calendar calendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
        calendar.setTime(new Date());
        calendar.add(Calendar.MINUTE, getCountdownTime());
        absoluteTime = calendar.getTime();
    }

    public Long getMillisToExpire() {
        Calendar calendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
        calendar.setTime(new Date());
        return absoluteTime.getTime() - calendar.getTimeInMillis();
    }
}
