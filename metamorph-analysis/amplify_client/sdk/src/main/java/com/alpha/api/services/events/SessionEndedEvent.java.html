package com.alpha.api.services.events;

public abstract class SessionEndedEvent {

    public static enum Reason {
        TEACHER_INITIATED,
        TEACHER_DISCONNECTED,
        CONNECTION_LOST
    }

    private final Reason reason;

    public SessionEndedEvent(Reason reason) {
        this.reason = reason;
    }

    public boolean isTeacherInitiated() {
        return reason == Reason.TEACHER_INITIATED;
    }

    public boolean isConnectionLost() {
        return reason == Reason.CONNECTION_LOST;
    }

    public boolean isTeacherDisconnected() {
        return reason == Reason.TEACHER_DISCONNECTED;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }

        SessionEndedEvent that = (SessionEndedEvent) o;

        return reason == that.reason;

    }

    @Override
    public int hashCode() {
        if (reason == null) {
            return 0;
        }
        return reason.hashCode();
    }
}
