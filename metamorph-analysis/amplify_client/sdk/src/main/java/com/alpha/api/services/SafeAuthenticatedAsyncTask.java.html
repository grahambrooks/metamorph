package com.alpha.api.services;

import com.alpha.api.authentication.AuthenticationException;
import com.alpha.logging.AnalyticsAgent;

public abstract class SafeAuthenticatedAsyncTask extends SafeTimedAsyncTask {
    public SafeAuthenticatedAsyncTask(AnalyticsAgent analyticsAgent) {
        super(analyticsAgent);
    }

    @Override
    protected final void onException(Exception e) {
        if (e.getClass() == AuthenticationException.class) {
            onAuthenticationException((AuthenticationException) e);
        } else {
            onOtherException(e);
        }
    }

    protected abstract void onAuthenticationException(AuthenticationException e);

    protected abstract void onOtherException(Exception e);
}
