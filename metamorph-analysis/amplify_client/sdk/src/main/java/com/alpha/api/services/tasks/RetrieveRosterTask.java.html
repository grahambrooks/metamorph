package com.alpha.api.services.tasks;

import com.alpha.api.authentication.AuthenticationException;
import com.alpha.api.services.SafeAuthenticatedAsyncTask;
import com.alpha.api.services.UserService;
import com.alpha.logging.AnalyticsAgent;
import com.alpha.logging.LogUtils;
import com.google.inject.Inject;
import org.slf4j.Logger;

public class RetrieveRosterTask extends SafeAuthenticatedAsyncTask {

    private static final Logger LOG = LogUtils.getLogger();

    private final UserService userService;

    @Inject
    public RetrieveRosterTask(UserService userService, AnalyticsAgent analyticsAgent) {
        super(analyticsAgent);
        this.userService = userService;
    }

    @Override
    public Void call() throws Exception {
        if (!userService.hasUserRegistered()) {
            LOG.trace("RetrieveRosterTask called without registered user");
            return null;
        }

        LOG.error("Retrieving updated student roster information");
        return null;
    }

    @Override
    protected void onAuthenticationException(AuthenticationException e) {
        LOG.error("Unexpected authentication exception ", e);
    }

    @Override
    protected void onOtherException(Exception e) {
        LOG.error("Unexpected exception occurred ", e);
    }

}
