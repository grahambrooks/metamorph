package com.alpha.api.domains;

import android.view.ViewGroup;
import org.apache.commons.lang3.Validate;

public abstract class LearningElementVisitor {

    public abstract T visit(VideoLearningElement element, ViewGroup parent);

    public abstract T visit(GoogleDriveLearningElement element, ViewGroup parent);

    public abstract T visit(WebPageLearningElement element, ViewGroup parent);

    public abstract T visit(EReaderLearningElement element, ViewGroup parent);

    public abstract T visit(GoogleMapsLearningElement element, ViewGroup parent);

    // This cannot be final since we mock it out in tests!!!
    public T get(LearningElement element, ViewGroup parent) {
        Validate.notNull(element, "Cannot visit a null LearningElement!");
        return element.getView(this, parent);
    }
}
