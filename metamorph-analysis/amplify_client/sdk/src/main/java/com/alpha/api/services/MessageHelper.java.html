package com.alpha.api.services;

import android.os.Bundle;
import android.os.Message;
import android.os.RemoteException;
import com.alpha.logging.LogUtils;
import org.slf4j.Logger;

public class MessageHelper {

    private static Logger LOG = LogUtils.getLogger();

    public void sendReply(Message msg, Bundle returnData) {
        try {
            // Reply using the same 'what' code
            Message replyMsg = Message.obtain(null, msg.what);
            replyMsg.setData(returnData);
            msg.replyTo.send(replyMsg);
        } catch (RemoteException e) {
            LOG.error("Exception while sending a message to other process: " + e);
        }
    }

}
