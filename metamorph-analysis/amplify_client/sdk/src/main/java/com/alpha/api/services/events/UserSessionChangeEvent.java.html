package com.alpha.api.services.events;

import com.alpha.api.services.Participant;

public class UserSessionChangeEvent {
    public static enum Type {
        JOIN,
        LEAVE
    }
    private final Participant participant;
    private final Type type;

    public UserSessionChangeEvent(Participant participant, Type type) {
        this.participant = participant;
        this.type = type;
    }

    public Participant getParticipant() {
        return participant;
    }

    public Type getType() {
        return type;
    }

    public boolean isUserJoining() {
        return Type.JOIN.equals(type);
    }
