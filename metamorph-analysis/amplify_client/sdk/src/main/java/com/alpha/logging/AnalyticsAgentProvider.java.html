package com.alpha.logging;

import com.alpha.api.properties.Environment;
import com.alpha.api.services.UserService;
import com.google.inject.Inject;
import com.google.inject.Provider;

import java.util.Collection;

import static com.google.common.collect.Lists.newArrayList;

public class AnalyticsAgentProvider implements Provider {

    private final UserService userService;
    private final boolean consoleAnalytics;

    @Inject
    public AnalyticsAgentProvider(Environment environment, UserService userService) {
        this.consoleAnalytics = environment.getConsoleAnalytics();
        this.userService = userService;
    }

    private static AnalyticsAgent build(UserService userService, boolean consoleAnalytics) {
        Collection adapters = newArrayList();
        if (consoleAnalytics) {
            adapters.add(new ConsoleAnalyticsAdapter());
        }
        return new AmplifyAnalyticsAgent(adapters, userService);
    }

    @Override
    public AnalyticsAgent get() {
        return build(userService, consoleAnalytics);
    }
}
