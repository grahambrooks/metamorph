package com.alpha.api.services;

import com.alpha.api.authentication.AuthenticationException;
import com.alpha.api.domains.Post;
import com.alpha.api.messaging.QuickActivitySettings;
import com.alpha.api.properties.Environment;
import com.alpha.api.services.callbacks.PostResponseCallback;
import com.alpha.api.services.web.WebServices;
import com.alpha.logging.AnalyticsAgent;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;

public class AddQuickActivityPostTask extends AddPostTask {
    private final QuickActivitySettings settings;

    @Inject
    public AddQuickActivityPostTask(WebServices webServices, Environment environment, AnalyticsAgent analyticsAgent, @Assisted Post post, @Assisted PostResponseCallback callback, @Assisted QuickActivitySettings settings) {
        super(webServices, environment, analyticsAgent, post, callback);
        this.settings = settings;
    }

    @Override
    public Post call() {
        settings.setRunning(true);
        return super.call();
    }

    @Override
    protected void onAuthenticationException(AuthenticationException e) {
        settings.setRunning(false);
        super.onAuthenticationException(e);
    }

    @Override
    protected void onOtherException(Exception e) {
        settings.setRunning(false);
        super.onOtherException(e);
    }

    @Override
    protected void onSuccess(Post post) {
        settings.setPost(post);
        super.onSuccess(post);
    }
}
