package com.alpha.api.services;

import android.app.Service;
import android.content.ServiceConnection;

public abstract class SundaeServiceConnection implements ServiceConnection {
    /**
     * Implementations should bind to the service that this ServiceConnection class
     * relates to and return the instance of the Service.
     * This should be performed through the ApplicationContext. If the binding
     * is unsuccessful then a RuntimeException should be thrown.
     */
    public abstract void open();

    /**
     * Implementations should unbind the Service that was connected to when open was called. This should
     * also be performed through the ApplicationContext.
     *
     */
    public abstract void close();

    /**
     * Should return the instance of the Service obtained when open was called. If the service instance was
     * not obtained or open was not called, then this method should throw a RuntimeException.
     * @return Instance of the Service
     */
    public abstract T getService();

}
