package com.alpha.api.domains;

import android.content.Intent;
import android.content.pm.PackageManager;
import android.net.Uri;
import android.view.ViewGroup;
import org.apache.commons.lang3.StringUtils;


public class EReaderLearningElement extends LearningElement {
    public static final String KALDOR_PACKAGE = "com.kaldorgroup.amplify";
    public static final String KALDOR_APPNAME = "eReader";

    private final String book;
    private final Uri imageUri;
    private final String bookUri;

    public EReaderLearningElement(String title, String description, String book, Uri imageUri, String bookUri) {
        super(title, description, LearningElementType.EREADER, StringUtils.EMPTY);
        this.book = book;
        this.imageUri = imageUri;
        this.bookUri = bookUri;
    }

    public Uri getEReaderUri(){
        return imageUri;

    public String getBook() {
        return book;
    }

    public Intent getEReaderShareIntent() {
        Intent shareIntent = new Intent();
        shareIntent.putExtra(Intent.EXTRA_SUBJECT, getBook());
        shareIntent.putExtra(Intent.EXTRA_STREAM, getEReaderUri());
        shareIntent.putExtra(Intent.EXTRA_TEXT, getTitle());
        return shareIntent;
    }

    public Intent getLaunchIntent(PackageManager packageManager) {
        Intent launchIntent = packageManager.getLaunchIntentForPackage(KALDOR_PACKAGE);
        launchIntent.setAction(Intent.ACTION_VIEW);
        launchIntent.setData(Uri.parse(bookUri));
        return launchIntent;
    }
