package com.alpha.api.authentication;

import com.google.common.cache.Cache;
import com.google.common.cache.CacheBuilder;
import com.google.inject.Inject;
import com.google.inject.name.Named;

import java.util.concurrent.TimeUnit;

public class AuthTokenCache {
    private static final String AUTH_TOKEN_KEY = "AUTH_TOKEN";
    public static final String AUTH_TOKEN_CACHE_INTERVAL = "auth_token_cache_time";
    private final Cache cache;

    @Inject
    AuthTokenCache(@Named(AUTH_TOKEN_CACHE_INTERVAL) int expiryTimeInMinutes) {
        this.cache = CacheBuilder
                .newBuilder()
                .expireAfterWrite(expiryTimeInMinutes, TimeUnit.MINUTES)
                .build();
    }

    public String get() {
        return cache.getIfPresent(AUTH_TOKEN_KEY);
    }

    public void put(String token) {
        cache.put(AUTH_TOKEN_KEY, token);
    }

    public void invalidateGoogleAuthToken() {
        cache.invalidate(AUTH_TOKEN_KEY);
    }

}
