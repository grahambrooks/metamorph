package com.alpha.api.messaging.xmpp;

import org.jivesoftware.smack.ConnectionConfiguration;
import org.jivesoftware.smack.XMPPConnection;
import org.jivesoftware.smack.XMPPException;
import org.jivesoftware.smack.packet.Presence;

public class AmplifyXmppConnection extends XMPPConnection {

    public AmplifyXmppConnection(ConnectionConfiguration config) {
        super(config);
    }

    protected void turnOffAutoReconnection() {
       config.setReconnectionAllowed(false);
    }

    protected void turnOnAutoReconnection() {
        config.setReconnectionAllowed(true);
    }

    @Override
    public void connect() throws XMPPException {
        turnOnAutoReconnection();
        super.connect();
    }

    @Override
    public synchronized void disconnect(Presence unavailablePresence) {
        super.disconnect(unavailablePresence);
        turnOffAutoReconnection();
    }
}
