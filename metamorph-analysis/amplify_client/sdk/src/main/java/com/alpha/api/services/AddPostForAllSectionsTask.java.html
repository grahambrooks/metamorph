import java.util.List;

import static com.alpha.api.http.HttpClient.Response;
import static com.alpha.api.services.ApiErrorReason.INTERNAL_ERROR;

public class AddPostForAllSectionsTask extends SafeAuthenticatedAsyncTask {
    private static final Logger LOG = LogUtils.getLogger();

    private final Post post;
    private final List sectionIds;
    private final PostResponseCallback callback;
    private final WebServices webServices;
    private final Environment environment;

    @Inject
    public AddPostForAllSectionsTask(WebServices webServices,
                                     Environment environment,
                                     AnalyticsAgent analyticsAgent,
                                     @Assisted Post post,
                                     @Assisted List sectionIds,
                                     @Assisted PostResponseCallback callback) {
        super(analyticsAgent);
        this.environment = environment;
        this.post = post;
        this.sectionIds = sectionIds;
        this.callback = callback;
        this.webServices = webServices;
    }

    @Override
    public Post call() {
        LOG.debug("JSON with sections ids" + post.asRequestJson(sectionIds));
        ApiRequest request = ApiRequest.createPostRequest(environment.getPostsURI(), post.asRequestJson(sectionIds));
        Response response = webServices.post(request);
        return response.getBodyAs(Post.class);
    }
