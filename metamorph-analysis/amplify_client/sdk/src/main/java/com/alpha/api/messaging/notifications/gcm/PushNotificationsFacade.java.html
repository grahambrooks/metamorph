package com.alpha.api.messaging.notifications.gcm;

import android.content.Context;
import com.google.android.gcm.GCMRegistrar;
import com.google.inject.Inject;
import com.google.inject.Singleton;

@Singleton
public class PushNotificationsFacade {

    static final String GOOGLE_PROJECT_NUMBER = "88093561502";

    @Inject
    private Context context;

    public void register() {
        GCMRegistrar.checkDevice(context);
        GCMRegistrar.checkManifest(context);
        GCMRegistrar.register(context, GOOGLE_PROJECT_NUMBER);
    }

    public void unregister() {
        GCMRegistrar.unregister(context);
    }

    public String getCurrentRegistrationId() {
        return GCMRegistrar.getRegistrationId(context);
    }
}
