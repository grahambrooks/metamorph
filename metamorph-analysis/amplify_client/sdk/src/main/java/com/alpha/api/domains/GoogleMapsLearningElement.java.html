package com.alpha.api.domains;

import android.content.Intent;
import android.net.Uri;
import android.view.ViewGroup;
import org.apache.commons.lang3.StringUtils;

import java.util.ArrayList;
import java.util.Collections;

import static com.google.common.collect.Lists.newArrayList;

public class GoogleMapsLearningElement extends LearningElement {

    public static final String GOOGLE_MAPS_URI_QUERY = "geo:0,0?q=";
    public static final String GOOGLE_MAPS_PACKAGE = "com.google.android.apps.maps";
    public static final String GOOGLE_MAPS_APPNAME = "Google Maps";

    private final String addressInfo;

    public GoogleMapsLearningElement(String title, String addressInfo) {
        super(title, StringUtils.EMPTY, LearningElementType.MAPS, null);
        this.addressInfo = addressInfo;
    }

    public Intent getLaunchIntent() {
        ArrayList list = newArrayList();
        Collections.addAll(list, addressInfo.split("\n"));
        list.remove(list.size() - 1);

        String mapsUri = GOOGLE_MAPS_URI_QUERY + list.toString();
        Intent intent = new Intent(Intent.ACTION_VIEW, Uri.parse(mapsUri));
        intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        return intent;
    }

    public Intent getShareIntent() {
        Intent shareIntent = new Intent();
        shareIntent.putExtra(Intent.EXTRA_SUBJECT, getTitle());
        shareIntent.putExtra(Intent.EXTRA_TEXT, addressInfo);
        return shareIntent;
    }
