package com.alpha.api.services.tasks;

import com.alpha.logging.LogUtils;
import org.slf4j.Logger;

import javax.annotation.Nullable;
import java.util.Date;

public class TimestampChangeDetector {
    private final Logger LOG = LogUtils.getLogger();
    @Nullable
    private Date lastTimestamp = null;
    private int repeatedTimestampCount;

    public void reset() {
        lastTimestamp = null; //NOPMD
        repeatedTimestampCount = 0;
    }

    public boolean isChanging(Date timestamp) {
        if ((lastTimestamp == null && timestamp == null) || (lastTimestamp != null && lastTimestamp.equals(timestamp))) {
            repeatedTimestampCount++;
        } else {
            repeatedTimestampCount = 0;
        }
        lastTimestamp = timestamp;

        final boolean timestampIsChanging = repeatedTimestampCount < 2;
        if (!timestampIsChanging) {
            LOG.trace("{} timestamps repeated the value of the previous timestamp: {}", repeatedTimestampCount, lastTimestamp);
        }
        return timestampIsChanging;
    }
}
