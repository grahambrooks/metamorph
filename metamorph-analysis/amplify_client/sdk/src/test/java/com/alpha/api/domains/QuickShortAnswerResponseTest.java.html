package com.alpha.api.domains;

import android.content.ContentValues;
import com.alpha.api.repository.sqlite.TableColumn;
import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import java.util.Date;

import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;

@RunWith(AmplifySDKTestRunner.class)
public class QuickShortAnswerResponseTest {

    private static final long ID= 99l;
    private static final long POST_ID = 88l;

    @Test
    public void shouldReturnContentValues() {
        Date createdOnDate = new Date();
        PostAuthor author = new PostAuthor("George Micheal");
        QuickShortAnswerResponse response = new QuickShortAnswerResponse(ID, "text", author, createdOnDate);

        ContentValues contentValues = response.asValues(POST_ID);

        assertThat(contentValues.getAsLong(TableColumn._id.name()), is(ID));
        assertThat(contentValues.getAsString(TableColumn.response_type.name()), is(ResponseType.QUICK_SHORT_ANSWER.name()));
        assertThat(contentValues.getAsLong(TableColumn.post_id.name()), is(POST_ID));
        assertThat(contentValues.getAsString(TableColumn.content.name()), is(response.asJson()));
        assertThat(contentValues.getAsLong(TableColumn.created_on.name()), is(response.getCreatedOn().getTime()));
        assertThat(contentValues.getAsString(TableColumn.answer.name()), is(response.getAnswer()));
    }
}
