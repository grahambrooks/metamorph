package com.alpha.api.services.tasks;

import com.alpha.api.authentication.ApiRequest;
import com.alpha.api.json.AmplifyJsonHelper;
import com.alpha.api.properties.Environment;
import com.alpha.api.services.web.WebServices;
import org.junit.Test;
import org.mockito.Mock;

import java.util.Collection;

import static org.mockito.Mockito.verify;
import static org.mockito.Mockito.when;
import static org.mockito.MockitoAnnotations.initMocks;

public class AddPostParticipantsTaskTest {

    private Collection activeParticipants;
    @Mock
    private WebServices webServices;
    @Mock
    private Environment environment;

    public static final long POST_ID = 1L;
    public static final String URI = "URI";

    @Test
    public void shouldUpdatePost () throws Exception
    {
        initMocks(this);
        when(environment.getAddParticipantsToPostUri(POST_ID)).thenReturn(URI);
        AddPostParticipantsTask task = new AddPostParticipantsTask(webServices, environment, null, POST_ID, activeParticipants);

        String json = AmplifyJsonHelper.toJson(activeParticipants);
        ApiRequest request = ApiRequest.createPutRequest(URI, json);

        task.call();

        verify(webServices).put(request);
    }
}
