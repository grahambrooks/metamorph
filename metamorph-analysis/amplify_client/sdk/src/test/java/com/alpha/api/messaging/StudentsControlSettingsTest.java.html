package com.alpha.api.messaging;

import com.alpha.api.json.AmplifyJsonHelper;
import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static junit.framework.Assert.assertTrue;
import static org.hamcrest.Matchers.*;
import static org.junit.Assert.assertThat;

@RunWith(AmplifySDKTestRunner.class)
public class StudentsControlSettingsTest {

    @Test
    public void shouldSerializeCommentsCorrectly() {
        StudentsControlSettings studentsControlSettings = new StudentsControlSettings();

        String output = AmplifyJsonHelper.toJson(studentsControlSettings);

        assertThat(output.contains("\"isCommentsEnabled\":false"), is(true));
    }

    @Test
    public void shouldSerializeLockedMessageCorrectly() {
        StudentsControlSettings studentsControlSettings = new StudentsControlSettings();
        studentsControlSettings.setAsLocked();
        String output = AmplifyJsonHelper.toJson(studentsControlSettings);
        assertThat(output.contains("\"isLocked\":true"), is(true));
    }

    @Test
    public void shouldCreateAnImmutableCopy() {
        StudentsControlSettings settings = new StudentsControlSettings();

        StudentsControlSettings copy = new StudentsControlSettings(settings);

        assertThat(copy, not(sameInstance(settings)));

        assertThat(copy.isLocked(), is(settings.isLocked()));
        assertThat(copy.isCommentsEnabled(), is(settings.isCommentsEnabled()));
    }

    @Test
    public void shouldLockDeviceOnSetAsLocked(){
        StudentsControlSettings settings = new StudentsControlSettings ();
        settings.setAsLocked();
        assertTrue(settings.isLocked());
    }
}
