package com.alpha.api.domains;

import android.content.Intent;
import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;

@RunWith(AmplifySDKTestRunner.class)
public class GoogleMapsLearningElementTest {

    public static final String WEBLINK = GoogleMapsLearningElement.GOOGLE_MAPS_URI_QUERY + "weblink";
    public static final String GOOGLE_MAPS_ADDRESS = "name\nphone\naddress\n"+WEBLINK;
    private GoogleMapsLearningElement mapsLearningElement;

    @Before
    public void setup(){
        mapsLearningElement = new GoogleMapsLearningElement("title", GOOGLE_MAPS_ADDRESS);
    }

    @Test
    public void shouldDropURLInAddressInfoForIntent() {
        Intent launchIntent = mapsLearningElement.getLaunchIntent();
        String address = launchIntent.getData().toString();
        assertThat(address.contains(WEBLINK), is(false));
    }
}
