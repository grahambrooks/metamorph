package com.alpha.api.messaging;

import org.junit.Test;

import static org.hamcrest.Matchers.nullValue;
import static org.hamcrest.core.Is.is;
import static org.junit.Assert.assertThat;

public class CallOnSomeoneSettingsTest {

    public static final long CALLED_ON_PARTICIPANT_ID = 2L;

    @Test
    public void shouldBeFalseWhenInitialized() {
        CallOnSomeoneSettings settings = new CallOnSomeoneSettings();

        assertThat(settings.isSomeoneAlreadyBeingCalledOn(), is(false));
    }

    @Test
    public void shouldHaveDefaultIdWhenInitialized() {
        CallOnSomeoneSettings settings = new CallOnSomeoneSettings();

        assertThat(settings.getCalledOnParticipantId(), is(nullValue()));
    }

    @Test
    public void shouldBeTrueIfSomeoneHasAlreadyBeenCalledOn() {
        CallOnSomeoneSettings settings = new CallOnSomeoneSettings();

        settings.callOnParticipant(CALLED_ON_PARTICIPANT_ID);

        assertThat(settings.isSomeoneAlreadyBeingCalledOn(), is(true));
    }

    @Test
    public void shouldRememberIdOfCalledOnParticipant() {
        CallOnSomeoneSettings settings = new CallOnSomeoneSettings();

        settings.callOnParticipant(CALLED_ON_PARTICIPANT_ID);

        assertThat(settings.getCalledOnParticipantId(), is(CALLED_ON_PARTICIPANT_ID));
    }
}
