package com.alpha.api.services;

import com.alpha.api.messaging.ParticipantPresenceModel;
import com.alpha.api.messaging.xmpp.XmppStudentSession;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.alpha.api.services.web.WebServices;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.BDDMockito.given;
import static org.mockito.Mockito.verify;
import static org.mockito.Mockito.when;
import static org.mockito.MockitoAnnotations.initMocks;


@RunWith(AmplifySDKTestRunner.class)
public class UpdateStudentWhereaboutsTaskTest {

    @Mock
    private ParticipantPresenceModel model;
    @Mock
    private XmppStudentSession studentSession;
    @Mock
    private Student student;
    @Mock
    private WebServices webService;
    @Mock
    private Event event;
    private UpdateStudentWhereaboutsTask updateStudentWhereaboutsTask;

    @Before
    public void setup() {
        initMocks(this);
        given(student.getPresenceModel()).willReturn(model);
        updateStudentWhereaboutsTask = new UpdateStudentWhereaboutsTask(studentSession, webService, student);
    }

    @Test
    public void shouldSendModelToStudentSession() throws Exception {
        updateStudentWhereaboutsTask.call();
        verify(studentSession).updateStudentPresenceStatus(model);
    }

    @Test
    public void shouldLogLocationUpdateEvent() throws Exception {
        when(student.changeWhereaboutsEvent()).thenReturn(event);
        updateStudentWhereaboutsTask.call();
        verify(webService).send(event);
    }

}
