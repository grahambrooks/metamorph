package com.alpha.api.utilities;

import org.junit.Test;

import static org.hamcrest.Matchers.is;
import static org.hamcrest.Matchers.nullValue;
import static org.junit.Assert.assertThat;

public class ReflectionUtilsTest {

    public static final String[] STRINGS = new String[]{"10"};
    public static final int INT = 10;

    @Test
    public void shouldLoadInstanceFromClass() throws Exception {
        Class aClass = ReflectionUtils.forceInit(String.class);
        assertThat(aClass == String.class, is(true));
    }

    @Test
    public void shouldFindMatchingConstructorWithMultipleArguments() throws Exception {
        MyTestClass instance = ReflectionUtils.newInstanceOf(MyTestClass.class, new Object[]{STRINGS, INT});
        assertThat(instance.strings, is(STRINGS));
        assertThat(instance.count, is(INT));
    }

    @Test
    public void shouldFindMatchingDefaultConstructor() throws Exception {
        MyTestClass instance = ReflectionUtils.newInstanceOf(MyTestClass.class);
        assertThat(instance.strings, is(nullValue()));
        assertThat(instance.count, is(0));
    }

    @Test(expected = IllegalArgumentException.class)
    public void shouldNotCreateInstanceIfNoMatchingConstructor() throws Exception {
        ReflectionUtils.newInstanceOf(MyTestClass.class, new Object[]{STRINGS});
    }

    @Test
    public void shouldFindMatchingConstructorWithSingleArgument() throws Exception {
        MyTestClass instance = ReflectionUtils.newInstanceOf(MyTestClass.class, INT);
        assertThat(instance.strings, is(nullValue()));
        assertThat(instance.count, is(INT));
    }
}

class MyTestClass {
    public String[] strings;
    public int count;

    public MyTestClass() {  //NOPMD
    }

    protected MyTestClass(int count) {
        this.count = count;
    }

    protected MyTestClass(String[] strings, int count) {
        this.strings = strings;
        this.count = count;
    }
}
