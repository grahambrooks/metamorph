package com.alpha.api.messaging;

import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static junit.framework.Assert.assertFalse;
import static junit.framework.Assert.assertTrue;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.core.Is.is;
import static org.hamcrest.core.IsNull.nullValue;

@RunWith(AmplifySDKTestRunner.class)
public class ControlSettingsModelTest {

    @Test
    public void shouldLockDeviceOnSetAsLocked() {
        ControlSettingsModel settings = new ControlSettingsModel();
        settings.setAsLocked();
        assertTrue(settings.isLocked());
    }

    @Test
    public void shouldEnableCallOnSomeoneWhenSetWithValidParticipantId() {
        CallOnSomeoneSettings settings = new ControlSettingsModel().getCallOnSomeoneSettings();
        settings.callOnParticipant(100L);
        assertTrue(settings.isCallOnSomeoneEnabled());
        assertThat(settings.getCalledOnParticipantId(), is(100L));
    }

    @Test
    public void shouldResetCallOnSomeoneParticipantIdWhenCallOnSomeoneIsEnd() {
        CallOnSomeoneSettings settings = new ControlSettingsModel().getCallOnSomeoneSettings();
        settings.resetCallOnSomeone();
        assertFalse(settings.isCallOnSomeoneEnabled());
        assertThat(settings.getCalledOnParticipantId(), nullValue());
    }
}
