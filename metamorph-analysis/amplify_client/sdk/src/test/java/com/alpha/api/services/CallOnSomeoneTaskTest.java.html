package com.alpha.api.services;

import com.alpha.api.messaging.CallOnSomeoneSettings;
import com.alpha.api.messaging.ControlSettingsModel;
import com.alpha.api.messaging.TeacherSession;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.alpha.api.services.events.CallOnSomeoneFailedEvent;
import com.google.common.eventbus.EventBus;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.BDDMockito.given;
import static org.mockito.Matchers.isA;
import static org.mockito.Mockito.verify;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(AmplifySDKTestRunner.class)
public class CallOnSomeoneTaskTest {

    @Mock
    private UserService userService;
    @Mock
    private TeacherSession teacherSession;
    @Mock
    private User user;
    @Mock
    private ControlSettingsModel controlSettingsModel;
    @Mock
    private CallOnSomeoneSettings callOnSomeoneSettings;
    @Mock
    private EventBus eventBus;

    private CallOnSomeoneTask callOnSomeoneTask;
    private static final Long PARTICIPANT_ID = 1L;

    @Before
    public void setUp() {
        initMocks(this);
        given(userService.getUser()).willReturn(user);
        given(user.getCurrentControlSettingsModel()).willReturn(controlSettingsModel);
        given(callOnSomeoneSettings.getCalledOnParticipantId()).willReturn(PARTICIPANT_ID);
        given(controlSettingsModel.getCallOnSomeoneSettings()).willReturn(callOnSomeoneSettings);

        callOnSomeoneTask = new CallOnSomeoneTask(userService, teacherSession, eventBus);
    }

    @Test
    public void shouldSendCalledOnParticipantNameAndIdWhenExecuted() throws Exception {
        callOnSomeoneTask.call();

        verify(teacherSession).sendStudentControlSettings(controlSettingsModel);
    }

    @Test
    public void shouldPostCallOnSomeoneFailedEventOnException() {
        callOnSomeoneTask.onException(new RuntimeException());

        verify(eventBus).post(isA(CallOnSomeoneFailedEvent.class));
    }

    @Test
    public void shouldSetSettingsAsUnlockedWhenTaskIsCalled() throws Exception {
        callOnSomeoneTask.call();

        verify(controlSettingsModel).setAsUnlocked();
    }
}
