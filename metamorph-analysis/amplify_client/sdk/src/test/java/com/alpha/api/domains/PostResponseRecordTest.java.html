package com.alpha.api.domains;

import android.content.ContentValues;
import com.alpha.api.repository.sqlite.TableColumn;
import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import java.util.Date;

import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;
import static org.hamcrest.Matchers.notNullValue;

@RunWith(AmplifySDKTestRunner.class)
public class PostResponseRecordTest {

    private PostResponseRecord postResponseRecord;
    private static final Long POST_ID = 1l;
    private static final Date CREATED_ON = new Date(132133);
    private static final Long AUTHOR_ID = 3L;
    private static final String AUTHOR_NAME = "Julie";

    @Before
    public void setUp() {
        postResponseRecord = new PostResponseRecord(POST_ID, AUTHOR_ID, AUTHOR_NAME, CREATED_ON);
    }

    @Test
    public void shouldReturnObjectAsContentValues() throws Exception {
        ContentValues values = postResponseRecord.asValues();

        assertThat(values.getAsLong(TableColumn._id.name()), is(notNullValue()));
        assertThat(values.getAsLong(TableColumn.post_id.name()), is(POST_ID));
        assertThat(values.getAsLong(TableColumn.author_id.name()), is(AUTHOR_ID));
        assertThat(values.getAsString(TableColumn.author_name.name()), is(AUTHOR_NAME));
        assertThat(values.getAsLong(TableColumn.created_on.name()), is(CREATED_ON.getTime()));
    }
}
