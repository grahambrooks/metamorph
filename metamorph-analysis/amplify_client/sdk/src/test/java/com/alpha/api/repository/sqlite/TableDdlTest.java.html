package com.alpha.api.repository.sqlite;

import org.junit.Test;
import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;

import java.util.Arrays;
import java.util.Collection;

import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;
import static org.junit.runners.Parameterized.Parameters;

@RunWith(Parameterized.class)
public class TableDdlTest {

    private final Table table;
    private final String expectedDDL;

    @Parameters
    public static Collection data() {
        return Arrays.asList(new Object[][] {
                {Table.POSTS, "CREATE TABLE POSTS (_id INTEGER PRIMARY KEY, created_on INTEGER NOT NULL, modified_on INTEGER NOT NULL, section_id INTEGER NOT NULL, post_type TEXT NOT NULL, content TEXT NOT NULL)"},
                {Table.SECTIONS, "CREATE TABLE SECTIONS (_id INTEGER PRIMARY KEY, name TEXT NOT NULL, course_id INTEGER NOT NULL, group_email TEXT)"},
                {Table.RESPONSES, "CREATE TABLE RESPONSES (_id INTEGER PRIMARY KEY, created_on INTEGER NOT NULL, post_id INTEGER NOT NULL, learning_element_id INTEGER, question_id INTEGER, selected_option_id INTEGER, response_type TEXT, answer TEXT, author_id INTEGER NOT NULL, quick_poll_answer TEXT, spot_check_answer TEXT, content TEXT NOT NULL)"},
                {Table.COMMENTS, "CREATE TABLE COMMENTS (_id INTEGER PRIMARY KEY, post_id INTEGER NOT NULL, created_on INTEGER NOT NULL, content TEXT NOT NULL)"},
                {Table.PARTICIPANTS, "CREATE TABLE PARTICIPANTS (_id INTEGER PRIMARY KEY, section_id INTEGER NOT NULL, participant_id INTEGER NOT NULL, name TEXT NOT NULL)"},
        });
    }
    public TableDdlTest(Table table, String expectedDDL) {
        this.table = table;
        this.expectedDDL = expectedDDL;
    }

    @Test
    public void shouldCreateCorrectDDLForTable() {
        assertThat(table.ddl(), is(expectedDDL));
    }
}
