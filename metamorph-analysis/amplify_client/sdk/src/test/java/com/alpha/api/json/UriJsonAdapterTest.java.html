package com.alpha.api.json;

import android.net.Uri;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.google.gson.JsonElement;
import com.google.gson.JsonPrimitive;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.containsString;
import static org.hamcrest.Matchers.is;

@RunWith(AmplifySDKTestRunner.class)
public class UriJsonAdapterTest {
    public static final Uri GOOGLE_URI = Uri.parse("http://google.com");
    private UriJsonAdapter jsonAdapter;

    @Test
    public void shouldSerializeUriToString() throws Exception {
        jsonAdapter = new UriJsonAdapter();
        JsonElement serializedElement = jsonAdapter.serialize(GOOGLE_URI, null, null);

        assertThat(serializedElement.getAsString(), containsString("http://google.com"));
    }

    @Test
    public void shouldDeSerializeUriToString() throws Exception {
        jsonAdapter = new UriJsonAdapter();
        Uri uri = jsonAdapter.deserialize(new JsonPrimitive("http://google.com"), null, null);

        assertThat(uri, is(GOOGLE_URI));
    }
}
