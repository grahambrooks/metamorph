package com.alpha.api.domains;

import android.content.ContentValues;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.alpha.api.services.Participant;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import java.util.Collection;

import static com.google.common.collect.Lists.newArrayList;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;

@RunWith(AmplifySDKTestRunner.class)
public class SectionUserTest {

    public static final long ID1 = 1L;
    public static final long ID2 = 2L;
    public static final String NAME1 = "P1";
    public static final String NAME2 = "P2";
    private SectionUser sectionUser;

    @Before
    public void setUp() throws Exception {
        Participant participant1 = new Participant(ID1, NAME1);
        Participant participant2 = new Participant(ID2, NAME2);
        sectionUser = new SectionUser(ID1, newArrayList(participant1, participant2));
    }

    @Test
    public void shouldTransformAllStudentsToContentValues() throws Exception {
        Collection contentValues = sectionUser.asContentValues();

        assertThat(contentValues.size(), is(2));
    }

}
