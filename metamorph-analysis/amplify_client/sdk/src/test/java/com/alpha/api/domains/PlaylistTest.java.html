package com.alpha.api.domains;

import android.content.ContentValues;
import android.net.Uri;
import com.alpha.api.repository.sqlite.TableColumn;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.alpha.api.services.User;
import com.alpha.api.services.UserRole;
import com.alpha.lock.GlobalInformationLock;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static com.google.common.collect.Lists.newArrayList;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(AmplifySDKTestRunner.class)
public class PlaylistTest {
    private static final Long PLAYLIST_ID = 1L;
    private static final String PLAYLIST_NAME = "title";
    private static final Long USER_ID = 1L;

    private Playlist playlist;
    @Mock
    private GlobalInformationLock globalInformationLock;

    @Before
    public void setUp() throws Exception {
        initMocks(this);

        User user = new User(USER_ID, PLAYLIST_NAME, UserRole.TEACHER, null, null, null, null,globalInformationLock);
        LearningElement learningElement = new WebPageLearningElement("Title", "Description", Uri.EMPTY);

        playlist = new Playlist(PLAYLIST_ID, PLAYLIST_NAME, user, newArrayList(learningElement));
    }

    @Test
    public void shouldPopulateContentValuesWithProperValues() {
        ContentValues values = playlist.asValues();

        assertThat(values.getAsString(TableColumn.content.name()), is(playlist.asJson()));
        assertThat(values.getAsLong(TableColumn._id.name()), is(PLAYLIST_ID));
        assertThat(values.getAsString(TableColumn.name.name()), is(PLAYLIST_NAME));
        assertThat(values.getAsLong(TableColumn.user_id.name()), is(USER_ID));
    }
}
