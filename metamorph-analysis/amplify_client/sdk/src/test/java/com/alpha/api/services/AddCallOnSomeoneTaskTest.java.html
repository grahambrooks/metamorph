package com.alpha.api.services;

import com.alpha.api.authentication.ApiRequest;
import com.alpha.api.properties.Environment;
import com.alpha.api.runner.AmplifySDKTestRunner;
import com.alpha.api.services.web.WebServices;
import com.alpha.logging.AnalyticsAgent;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.Mockito.verify;
import static org.mockito.Mockito.when;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(AmplifySDKTestRunner.class)
public class AddCallOnSomeoneTaskTest {

    public static final String URL = "url";
    public static final String JSON = "{\"userId\":1}";
    private static final Long POST_ID = 1L;
    private static final Long USER_ID = 1L;

    private AddCallOnSomeoneTask task;

    @Mock
    private WebServices webServices;
    @Mock
    private Environment environment;
    @Mock
    private AnalyticsAgent analyticsAgent;

    @Before
    public void setUp() throws Exception {
        initMocks(this);
        when(environment.getCallOnSomeOneURI(POST_ID, USER_ID)).thenReturn(URL);
        task = new AddCallOnSomeoneTask(webServices, environment, analyticsAgent, POST_ID, USER_ID);
    }

    @Test
    public void shouldCallPostMethodOnWebServiceWhenTaskGetExecuted() throws Exception {
        task.call();

        ApiRequest apiRequest = ApiRequest.createPostRequest(URL, JSON);

        verify(webServices).post(apiRequest);
    }
}
