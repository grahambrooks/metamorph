package com.alpha.api.services;


import com.alpha.lock.GlobalInformationLock;
import com.google.common.collect.Lists;
import org.junit.Before;
import org.junit.Test;

import static org.mockito.Matchers.isA;
import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.verify;

public class UserLockingTest {
    private SynchronisedUser synchronisedUser;
    private GlobalInformationLock globalInfoLock;


    @Before
    public void setUp() {
        globalInfoLock = mock(GlobalInformationLock.class);
        synchronisedUser = new SynchronisedUser(Lists.
newArrayList(), "", globalInfoLock) {};
    }


    @Test
    public void shouldCallWithReadLockWhenGettingSections(){
        synchronisedUser.getSections();
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }

    @Test
    public void shouldCallWithReadLockWhenGettingCurrentSection(){
        synchronisedUser.getCurrentSection();
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }

    @Test
    public void shouldCallWithReadLockWhenGettingSectionWithId(){
        synchronisedUser.getSectionWithId(1L);
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }

    @Test
    public void shouldCallWithReadLockWhenSettingCurrentSection(){
        synchronisedUser.setCurrentSection(1L);
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }

    @Test
    public void shouldCallWithReadLockWhenGettingName(){
        synchronisedUser.getName();
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }

    @Test
    public void shouldCallWithReadLockWhenSettingCurrentSectionToNone(){
        synchronisedUser.setCurrentSectionToNone();
        verify(globalInfoLock).callWithReadLock(isA(AmplifyCallable.class));
    }
}
