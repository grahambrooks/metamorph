package com.alpha.api.repository.contentprovider;

import android.content.ContentValues;
import android.database.sqlite.SQLiteDatabase;
import android.net.Uri;
import com.alpha.api.repository.sqlite.SundaeDatabase;
import com.alpha.api.runner.AmplifySDKTestRunner;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.BDDMockito.given;
import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.verify;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(AmplifySDKTestRunner.class)
public class SundaeContentProviderBulkInsertTest {

    private SundaeContentProvider provider;
    @Mock
    private SQLiteDatabase writableDatabase;
    @Mock
    private Uri uri;

    @Before
    public void setUp() throws Exception {
        initMocks(this);
        SundaeDatabase database = mock(SundaeDatabase.class);
        given(database.getWritableDatabase()).willReturn(writableDatabase);
        provider = new SundaeContentProvider();
        provider.dbHelper = database;
    }

    @Test
    public void shouldWrapInsertsInTransaction() throws Exception {
        provider.bulkInsert(uri, new ContentValues[0]);

        verify(writableDatabase).beginTransaction();
        verify(writableDatabase).setTransactionSuccessful();
        verify(writableDatabase).endTransaction();
    }
}
