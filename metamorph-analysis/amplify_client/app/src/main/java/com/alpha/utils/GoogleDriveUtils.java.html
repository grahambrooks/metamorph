package com.alpha.utils;

import com.google.inject.Singleton;

import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import static com.google.common.collect.Lists.newArrayList;

@Singleton
public class GoogleDriveUtils {
    public static final Pattern DOCUMENT_PATTERN = Pattern.compile(".*\\/d\\/(.*)\\/.*");
    private static final Pattern SPREADSHEET_PATTERN = Pattern.compile(".*key=(.*)#.*");
    private static final Pattern SPREADSHEET_PATTERN_WITHOUT_GID = Pattern.compile(".*key=(.*)");

    private static final List PATTERNS = newArrayList(DOCUMENT_PATTERN, SPREADSHEET_PATTERN, SPREADSHEET_PATTERN_WITHOUT_GID);
    public static final int MATCHED_GROUP = 1;

    public String getFileId(String url) {
        for (Pattern pattern : PATTERNS) {
            final Matcher matcher = pattern.matcher(url);
        if (matcher.matches()) {
                return matcher.group(MATCHED_GROUP);
            }
        }
        return null;
    }

    public boolean isGDoc(String url){
        return url != null && getFileId(url) != null;
    }
}
