package com.alpha.widgets.googledrive;

import com.alpha.activities.callable.GetFileInformation;
import com.google.inject.Inject;

public class GoogleFileLoader implements FileLoader {

    private final GoogleDriveProvider provider;

    @Inject
    public GoogleFileLoader(GoogleDriveProvider provider) {
        this.provider = provider;
    }

    @Override
    public void load(String fileId, LoadAction action) {
        new OnContentLoaded(new GetFileInformation(provider, fileId)).register(action);
    }
}
