package com.alpha.activities;

import android.content.Context;
import android.database.Cursor;

import static com.alpha.api.repository.sqlite.Order.ASC;
import static com.alpha.api.repository.sqlite.Table.COMMENTS;
import static com.alpha.api.repository.sqlite.TableColumn.*;

public class CommentCursorLoader extends BaseCursorLoader {

    private static final String[] PROJECTION = new String[]{_id.name(), created_on.name(), content.name()};
    private static final String SELECTION = post_id.equalTo();
    private static final String SORT_ORDER = ASC.by(created_on);

    public CommentCursorLoader(Context context, PostCursorLoaderCallbacks callbacks, Long postId) {
        super(context, COMMENTS.uri(postId), PROJECTION, SELECTION, new String[]{postId.toString()}, SORT_ORDER, callbacks, postId);
    }

    @Override
    public void onLoadFinished(Cursor cursor) {
        callbacks.setCommentsCursor(cursor);
    }

    @Override
    public void onLoaderReset() {
        callbacks.setCommentsCursor(null);
    }
}
