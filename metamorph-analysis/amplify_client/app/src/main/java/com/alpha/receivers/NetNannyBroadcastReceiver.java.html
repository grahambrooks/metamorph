package com.alpha.receivers;

import android.content.Context;
import android.content.Intent;
import com.alpha.api.services.events.NetNannyConfigurationEvent;
import com.alpha.logging.LogUtils;
import com.google.common.eventbus.EventBus;
import com.google.inject.Inject;
import org.slf4j.Logger;
import roboguice.receiver.RoboBroadcastReceiver;

public class NetNannyBroadcastReceiver extends RoboBroadcastReceiver {
    public static final Logger LOGGER = LogUtils.getLogger();
    public static final String NETNANNY_STATUS_KEY = "status";

    @Inject
    private EventBus eventBus;

    @Override
    protected void handleReceive(Context context, Intent intent) {
        boolean status = intent.getBooleanExtra(NETNANNY_STATUS_KEY, false);

        if(status) {
            eventBus.post(new NetNannyConfigurationEvent(true));
        }else {
            eventBus.post(new NetNannyConfigurationEvent(false));
        }
        LOGGER.debug("Received NetNanny setup status: " + status);
    }
}
