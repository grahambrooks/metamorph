package com.alpha.activities;

import android.app.Activity;
import android.app.FragmentManager;
import android.app.FragmentTransaction;

import javax.inject.Inject;

import com.amplify.R;

public class NotebookStreamFragmentManager {
    private final Activity context;

    @Inject
    public NotebookStreamFragmentManager(Activity context) {
        this.context = context;
    }

    public void loadNotebookStreamFragment() {
        FragmentManager fragmentManager = context.getFragmentManager();
        FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();
        fragmentTransaction.add(R.id.stream_fragment_layout, new NotebookStreamFragment(), NotebookStreamFragment.TAG);
        fragmentTransaction.commit();
    }

}
