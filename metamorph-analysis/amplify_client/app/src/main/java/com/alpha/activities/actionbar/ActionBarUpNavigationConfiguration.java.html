package com.alpha.activities.actionbar;

import android.app.ActionBar;
import android.app.Activity;
import android.widget.ImageView;
import com.google.inject.Inject;

public class ActionBarUpNavigationConfiguration {

    private final Activity activity;

    @Inject
    public ActionBarUpNavigationConfiguration(Activity activity) {
        this.activity = activity;
    }

    public void setIconForUpNavigation(int resourceId) {
        ActionBar actionBar = activity.getActionBar();

        actionBar.setDisplayShowHomeEnabled(true);
        actionBar.setDisplayHomeAsUpEnabled(true);
        actionBar.setDisplayShowTitleEnabled(false);

        activity.getActionBar().setIcon(resourceId);

        // TOM: we have to style the home icon in code since it is a default android layout that we do not control
        ImageView homeView = (ImageView) activity.findViewById(android.R.id.home);
        if (homeView != null) {
            // Android does not return null but Robolectric does in tests!
            homeView.setPadding(5, 0, 10, 0);
        }
    }

}
