package com.alpha.activities;

import java.util.Arrays;
import java.util.List;

public class QuickPollStudentResponseActivity extends LockedInClassActivity {
    @InjectView(R.id.quick_poll_choice_a)
    private View choiceA;

    @InjectView(R.id.quick_poll_choice_b)
    private View choiceB;

    @InjectView(R.id.quick_poll_choice_c)
    private View choiceC;

    @InjectView(R.id.quick_poll_choice_d)
    private View choiceD;

    @InjectFragment(R.id.network_status_quick_poll_response)
    NetworkStatusFragment networkStatus;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.quick_poll_student_response);
        setStudentSelectedResponse();
    }

    private void setStudentSelectedResponse() {
        String studentChoice = ((QuickPollSettings)getActivitySettings()).getStudentChoice();
        List choices = Arrays.asList(choiceA, choiceB, choiceC, choiceD);
        for (View choice : choices) {
            if (ObjectUtils.equals(studentChoice, choice.getTag())) {
                choice.setVisibility(View.VISIBLE);
            }
        }
    }
