package com.alpha.activities.adapters;

import android.content.ContentResolver;
import android.content.Context;
import android.database.Cursor;
import android.widget.CursorAdapter;
import android.widget.SimpleCursorAdapter;
import com.alpha.api.repository.sqlite.Order;
import com.alpha.api.repository.sqlite.Table;
import com.alpha.api.repository.sqlite.TableColumn;
import com.amplify.R;

import static com.alpha.api.repository.sqlite.TableColumn._id;
import static com.alpha.api.repository.sqlite.TableColumn.asStrings;
import static com.alpha.api.repository.sqlite.TableColumn.name;

public class CoursePlaylistsCursorAdapter extends SimpleCursorAdapter {

    private static final String[] FROM = asStrings(TableColumn.name);
    private static final int[] TO = new int[]{R.id.library_course_name};

    public CoursePlaylistsCursorAdapter(Context context) {
        super(context, R.layout.library_course_row, getCursor(context.getContentResolver()), FROM, TO, CursorAdapter.FLAG_REGISTER_CONTENT_OBSERVER);
    }

    private static Cursor getCursor(ContentResolver contentResolver) {
        return contentResolver.query(Table.COURSES.uri(), asStrings(_id, name), "", new String[]{}, Order.ASC.by(name));
    }

}
