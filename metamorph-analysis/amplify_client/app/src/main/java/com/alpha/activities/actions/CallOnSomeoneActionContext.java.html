package com.alpha.activities.actions;

import android.app.Activity;
import android.content.Context;
import com.alpha.activities.CallOnSomeoneActivity;
import com.alpha.api.services.UserService;
import com.alpha.api.services.factory.SectionTaskFactory;
import com.alpha.api.services.factory.SessionTaskFactory;
import com.amplify.R;
import roboguice.util.SafeAsyncTask;

public class CallOnSomeoneActionContext extends ActionContext {

    public CallOnSomeoneActionContext(Context context,
                                      SessionTaskFactory sessionTaskFactory,
                                      SectionTaskFactory sectionTaskFactory,
                                      UserService userService) {
        super(context, sessionTaskFactory, sectionTaskFactory, userService);
    }

    @Override
    public String getActionName() {
        return context.getString(R.string.call_on);
    }

    @Override
    public SafeAsyncTask getTaskToCompleteAction() {
        return sessionTaskFactory.createEndCallOnSomeoneTask();
    }

    @Override
    public Class activityType() {
        return CallOnSomeoneActivity.class;
    }
}
