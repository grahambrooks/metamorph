package com.alpha.activities.renderers;

import android.content.Context;
import android.view.View;
import android.widget.TextView;
import com.alpha.api.domains.*;
import com.amplify.R;

import java.text.MessageFormat;

public class LearningElementTitle {

    private final Context context;

    public LearningElementTitle(Context context) {
        this.context = context;
    }

    public TextView setText(View layout, VideoLearningElement element) {
        return setText(layout, element, R.string.learning_activity_header_video);
    }

    public TextView setText(View layout, GoogleDriveLearningElement element) {
        return setText(layout, element, R.string.learning_activity_header_gdrive);
    }

    private TextView setText(View layout, LearningElement element, int titleFormatId) {
        String titleFormat = context.getResources().getString(titleFormatId);
        TextView titleView = (TextView) layout.findViewById(R.id.learning_element_title);
        titleView.setText(MessageFormat.format(titleFormat, element.getTitle()));
        return titleView;
    }
}
