package com.alpha.application;

import android.app.Application;
import android.os.StrictMode;
import com.alpha.api.AlphaSDK;
import com.alpha.logging.LogUtils;
import org.slf4j.Logger;

public class AlphaApplication extends Application {

    public static final Logger LOGGER = LogUtils.getLogger();



    @Override
    public void onCreate() {
        super.onCreate();

        configureStrictMode();
        Thread.setDefaultUncaughtExceptionHandler(new AmplifyUncaughtHandler());

        AlphaSDK alphaSDK = new AlphaSDK(this);
        AlphaConfigurationModule alphaConfigurationModule = new AlphaConfigurationModule(alphaSDK);

        alphaSDK.initialise(alphaConfigurationModule);
    }

    private void configureStrictMode() {
        StrictMode.setThreadPolicy(new StrictMode.ThreadPolicy.Builder()
                .detectNetwork()
                .detectDiskWrites()
                .penaltyLog()
               // .penaltyFlashScreen() Hide it for temporarily for Production
                .build());
        StrictMode.setVmPolicy(new StrictMode.VmPolicy.Builder()
                .detectAll()
                .penaltyLog()
                .build());
    }




}
