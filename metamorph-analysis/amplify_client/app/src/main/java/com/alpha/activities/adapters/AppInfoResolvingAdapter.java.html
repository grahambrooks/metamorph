package com.alpha.activities.adapters;

import android.content.Context;
import android.content.Intent;
import android.content.pm.PackageManager;
import android.widget.ArrayAdapter;
import android.widget.TextView;
import com.alpha.api.domains.SundaeAppInfo;

import java.util.Set;

import static com.google.common.collect.Sets.newHashSet;

public class AppInfoResolvingAdapter extends ArrayAdapter {
    protected final PackageManager packageManager;

    public AppInfoResolvingAdapter(Context context, int resource, int textViewResourceId) {
        super(context, resource, textViewResourceId);
        packageManager = context.getPackageManager();
    }

    protected Intent launchActivityForPackage(String packageName) {
        return packageManager.getLaunchIntentForPackage(packageName);
    }

    protected void loadLabel(SundaeAppInfo sundaeAppInfo, TextView iconView) {
        iconView.setText(sundaeAppInfo.getAppLabel());
    }

    public Set getAllItems() {
        Set sundaeAppInfos = newHashSet();
        for (int i = 0; i < getCount(); i++) {
            final SundaeAppInfo item = getItem(i);
            if (!item.isAmplifyApp() && !item.isSectorApp()) {
                sundaeAppInfos.add(item);
            }
        }
        return sundaeAppInfos;
    }
}
