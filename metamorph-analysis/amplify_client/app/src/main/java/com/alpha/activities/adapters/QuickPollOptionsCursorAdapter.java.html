package com.alpha.activities.adapters;

import android.content.Context;
import android.view.View;
import android.widget.TextView;
import com.alpha.api.adapters.ContentJsonCursorAdapter;
import com.alpha.api.domains.Response;
import com.alpha.api.services.events.QuickPollResponseEvent;
import com.amplify.R;
import com.google.common.eventbus.EventBus;

public class QuickPollOptionsCursorAdapter extends ContentJsonCursorAdapter {
    private final EventBus eventBus;

    public QuickPollOptionsCursorAdapter(Context context, EventBus eventBus) {
        super(context, R.layout.quick_activity_students_list_item);
        this.eventBus = eventBus;
    }

    @Override
    protected void populateView(View view, Response response) {
        ((TextView) view).setText(response.getAuthor().getName());
        eventBus.post(new QuickPollResponseEvent(response.getAuthor()));
    }
}
