package com.alpha.utils;

import java.util.Locale;
import java.util.concurrent.TimeUnit;

public class CountdownFormatter {

    private CountdownFormatter() {
    }

    public static String formatRemainingTimeForTeacher(Long remainingTime) {
        long minute = TimeUnit.MILLISECONDS.toMinutes(remainingTime);
        long seconds = TimeUnit.MILLISECONDS.toSeconds(remainingTime) - (minute * 60);
        return format("%02d:%02d", minute, seconds);
    }

    public static String formatRemainingTimeForStudent(long remainingTime) {
        long minute = TimeUnit.MILLISECONDS.toMinutes(remainingTime);
        long seconds = TimeUnit.MILLISECONDS.toSeconds(remainingTime) - (minute * 60);
        // Help me please
        if (minute > 1) {
            return format("%s mins", minute);
        } else if (minute == 0) {
            return format("%s sec", seconds);
        } else {
            return "1 min";
        }
    }

    private static String format(String format, Object... args) {
        return String.format(Locale.getDefault(), format,  args);
    }
}
