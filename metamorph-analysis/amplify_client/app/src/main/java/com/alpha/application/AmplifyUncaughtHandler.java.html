package com.alpha.application;

import com.alpha.logging.LogUtils;
import org.slf4j.Logger;

import java.lang.Thread.UncaughtExceptionHandler;

class AmplifyUncaughtHandler implements UncaughtExceptionHandler {

    public static final Logger LOGGER = LogUtils.getLogger();
    private final UncaughtExceptionHandler delegate;

    AmplifyUncaughtHandler() {
        this(Thread.getDefaultUncaughtExceptionHandler());
    }

    AmplifyUncaughtHandler(UncaughtExceptionHandler delegate) {
        this.delegate = delegate;
    }

    @Override
    public void uncaughtException(Thread thread, Throwable ex) {
        LOGGER.error("Unexpected error happened in thread: '" + thread.getName() + ": " + thread.getId() + "'", ex);
        if (delegate != null) {
            delegate.uncaughtException(thread, ex);
        }
        exit();
    }

    protected void exit() {
        System.exit(-1);
    }
}
