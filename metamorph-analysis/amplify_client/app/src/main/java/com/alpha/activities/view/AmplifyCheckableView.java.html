package com.alpha.activities.view;

import android.content.Context;
import android.util.AttributeSet;
import android.widget.Checkable;
import android.widget.LinearLayout;
import com.amplify.R;

public class AmplifyCheckableView extends LinearLayout implements Checkable {
    public static final int[] STATE_CHECKED = new int[]{android.R.attr.state_checked};
    public static final int SPACE_FOR_NEW_FLAGS = 1;
    private Checkable checkable;

    public AmplifyCheckableView(Context context) {
        super(context);
    }

    public AmplifyCheckableView(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    public AmplifyCheckableView(Context context, AttributeSet attrs, int defStyle) {
        super(context, attrs, defStyle);
    }

    @Override
    protected void onFinishInflate() {
        super.onFinishInflate();
        checkable = (Checkable) findViewById(R.id.selector);
    }

    @Override
    protected int[] onCreateDrawableState(int extraSpace) {
        int[] state = super.onCreateDrawableState(extraSpace + SPACE_FOR_NEW_FLAGS);

        if (checkable != null && isChecked()) {
            mergeDrawableStates(state, STATE_CHECKED);
        }

        return state;
    }

    @Override
    public void setChecked(boolean b) {
        checkable.setChecked(b);
    }

    @Override
    public boolean isChecked() {
        return checkable.isChecked();
    }

    @Override
    public void toggle() {
        checkable.toggle();
        refreshDrawableState();
    }
}
