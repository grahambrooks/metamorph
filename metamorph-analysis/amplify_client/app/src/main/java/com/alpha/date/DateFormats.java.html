package com.alpha.date;

import org.ocpsoft.pretty.time.PrettyTime;

import java.util.Calendar;
import java.util.Date;
import java.util.Locale;

public class DateFormats {

    private static final String MONTH_YEAR_FORMAT = "%s %s";

    private DateFormats() {
    }

    public static String formatPastDateToDefaultStyle(Date date) {
        Date now = new Date();
        PrettyTime prettyTime = new PrettyTime(now);
        if ((date != null) && date.before(now)) {
            return prettyTime.format(date);
        } else {
            // force the date to be 'moments ago' rather than 'moments from now'
            Calendar cal = Calendar.getInstance();
            cal.setTime(now);
            cal.add(Calendar.SECOND, -5);
            return prettyTime.format(cal.getTime());
        }
    }

    public static String getCalendarMonthAndYear(Calendar calendar) {
        return String.format(MONTH_YEAR_FORMAT,
                calendar.getDisplayName(Calendar.MONTH, Calendar.LONG, Locale.getDefault()),
                calendar.get(Calendar.YEAR));
    }

    public static String getDayOfWeekLetter(Calendar calendar) {
        return calendar.getDisplayName(Calendar.DAY_OF_WEEK, Calendar.SHORT, Locale.getDefault()).substring(0, 1);
    }
}
