package com.alpha.activities.adapters;

import android.content.Context;
import android.view.View;
import android.widget.TextView;
import com.alpha.api.adapters.ContentJsonCursorAdapter;
import com.alpha.api.domains.Comment;
import com.alpha.date.DateFormats;
import com.amplify.R;
import com.google.inject.Inject;

public class DiscussionCommentsCursorAdapter extends ContentJsonCursorAdapter {

    private static final int[] AUTHOR_COLOR_LIST = new int[]{
            R.color.convo_color_green,
            R.color.convo_color_light_blue,
            R.color.convo_color_purple,
            R.color.convo_color_light_orange
    };

    @Inject
    public DiscussionCommentsCursorAdapter(Context context) {
        super(context, R.layout.discussion_comment_row);
    }

    @Override
    protected void populateView(View view, Comment comment) {
        final int position = getCursor().getPosition();

        final boolean isEvenRow = (position % 2) == 0;
        view.setBackgroundResource(isEvenRow ?
                R.drawable.discussion_comment_row_background_selector :
                R.drawable.discussion_comment_row_alternate_background_selector
        );

        final int colorIndex = position % AUTHOR_COLOR_LIST.length;
        final int color = getContext().getResources().getColor(AUTHOR_COLOR_LIST[colorIndex]);
        TextView authorText = (TextView) view.findViewById(R.id.comment_author);
        authorText.setTextColor(color);
        authorText.setText(comment.getAuthor().getName());

        TextView commentText = (TextView) view.findViewById(R.id.comment_content);
        commentText.setText(comment.getText());

        TextView createdDate = (TextView) view.findViewById(R.id.comment_timestamp);
        createdDate.setText(DateFormats.formatPastDateToDefaultStyle(comment.getCreatedOn()));
    }
}
