package com.alpha.activities.adapters;

import android.content.Context;
import android.database.Cursor;
import android.view.View;
import android.widget.SimpleCursorAdapter;
import android.widget.TextView;
import com.alpha.api.repository.sqlite.TableColumn;
import com.alpha.date.DateFormats;
import com.amplify.R;
import com.google.inject.Inject;

import java.util.Date;

public class PostResponseRecordsCursorAdapter extends SimpleCursorAdapter {

    public static final String RESPONSE_TIMESTAMP = "_response_timestamp";
    public static final String RESPONSE_NAME = "_response_name";

    @Inject
    public PostResponseRecordsCursorAdapter(Context context) {
        super(context, R.layout.response_name_and_timestamp_item, null, new String[] {}, new int[] {}, 0);
    }

    @Override
    public void bindView(View view, Context context, Cursor cursor) {
        view.setTag(retrieveLongValue(cursor, TableColumn.author_id));
        TextView authorText = (TextView) view.findViewById(R.id.response_author_column);
        String name = retrieveStringValue(cursor, TableColumn.author_name);
        authorText.setText(name);
        authorText.setContentDescription(name + RESPONSE_NAME);

        TextView createdDate = (TextView) view.findViewById(R.id.response_timestamp_column);
        createdDate.setText(retrieveDateValue(cursor, TableColumn.created_on));
        createdDate.setContentDescription(name + RESPONSE_TIMESTAMP);
    }

    private long retrieveLongValue(Cursor cursor, TableColumn column) {
        return cursor.getLong(cursor.getColumnIndex(column.name()));
    }

    private String retrieveDateValue(Cursor cursor, TableColumn tableColumn) {
        return DateFormats.formatPastDateToDefaultStyle(new Date(cursor.getLong(cursor.getColumnIndex(tableColumn.name()))));
    }
