package com.alpha.activities;

import android.app.LoaderManager;
import android.content.Context;
import android.content.Loader;
import android.database.Cursor;
import android.os.Bundle;
import com.alpha.activities.adapters.SpotCheckResponseCursorAdapter;
import com.alpha.api.adapters.ResponsesCursorLoader;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;

import static com.alpha.api.repository.sqlite.TableColumn.spot_check_answer;

public class SpotCheckResponseLoaderInitializer implements LoaderManager.LoaderCallbacks {

    protected static final Integer SPOT_CHECK_RESPONSE_LOADER_ID = 49;
    private final Context context;
    private final Long postId;
    private final SpotCheckResponseCursorAdapter spotCheckResponseCursorAdapter;

    @Inject
    public SpotCheckResponseLoaderInitializer(Context context,
                                              @Assisted Long postId,
                                              @Assisted SpotCheckResponseCursorAdapter spotCheckOptionsCursorAdapter) {
        this.context = context;
        this.postId = postId;
        this.spotCheckResponseCursorAdapter = spotCheckOptionsCursorAdapter;
    }

    @Override
    public Loader onCreateLoader(int id, Bundle args) {
        return new ResponsesCursorLoader(context, postId, spot_check_answer);
    }

    @Override
    public void onLoadFinished(Loader loader, Cursor cursor) {
        spotCheckResponseCursorAdapter.swapCursor(cursor);
    }

    @Override
    public void onLoaderReset(Loader loader) {
        spotCheckResponseCursorAdapter.swapCursor(null);
    }

    public void register(LoaderManager loaderManager) {
        loaderManager.initLoader(SPOT_CHECK_RESPONSE_LOADER_ID, null, this);
    }
}
