package com.alpha.utils;

public class NumberPicker {

    public static final int MIN_COUNT_DOWN_VALUE = 1;
    public static final int MAX_COUNT_DOWN_VALUE = 20;
    public static final int DEFAULT_TIMER_VALUE = 5;

    private final int min;
    private final int max;
    private int current;

    public NumberPicker(int min, int max) {
        this(min, max, DEFAULT_TIMER_VALUE);
    }

    public NumberPicker(int min, int max, int current) {
        this.current = current;
        this.min = min;
        this.max = max;
    }

    public int increment() {
        return current < max ? ++current : max;
    }

    public int decrement() {
        return current <= min ? min : --current;
    }

    public int getCurrent() {
        return current;
    }

    public void setCurrent(int current) {
        this.current = isOutOfRange(current) ? this.current : current;
    }

    private boolean isOutOfRange(int currentValue) {
        return (currentValue > this.max || currentValue < this.min);
    }
}
