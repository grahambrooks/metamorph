package com.alpha.activities.presenters;

import android.content.Context;
import android.content.res.Resources;
import android.view.View;
import android.widget.TextView;
import com.alpha.api.services.Section;
import com.alpha.api.services.UserService;
import com.amplify.R;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;

import java.text.MessageFormat;

import static com.alpha.utils.CharSequences.highlight;

public class MonitorViewPresenter {

    private final Resources resources;
    private final TextView studentsInSessionCount;
    private final UserService userService;

    @Inject
    public MonitorViewPresenter(@Assisted View view, Context context, UserService userService) {
        this.studentsInSessionCount = (TextView) view.findViewById(R.id.students_in_session_count);
        this.resources = context.getResources();
        this.userService = userService;
    }

    public void setViewToOutOfSession() {
        updateCounts(false);
    }

    public void setViewToInSession() {
        updateCounts(true);
    }

    public void updateSessionCounts(boolean inSession) {
        updateCounts(inSession);
    }

    private void updateCounts(boolean inSession) {
        String inSessionCount;
        Section currentUserSection = userService.getCurrentUserSection();

        if (inSession) {
            inSessionCount = MessageFormat.format(resources.getString(R.string.students_in_session_count),
                    currentUserSection.getInSessionParticipantsCount(),
                    currentUserSection.getAllParticipantsCount());
        } else {
            inSessionCount = resources.getString(R.string.no_class_in_session_count);
        }

        studentsInSessionCount.setText(highlight(inSessionCount, resources, R.color.primary_orange));
    }
}
