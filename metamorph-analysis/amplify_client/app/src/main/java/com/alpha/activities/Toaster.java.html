package com.alpha.activities;

import android.content.Context;
import android.widget.Toast;
import com.amplify.R;
import com.alpha.api.services.ApiErrorReason;
import com.google.common.collect.Maps;
import com.google.inject.Inject;
import com.google.inject.Singleton;

import java.util.Map;

@Singleton
public class Toaster {

    private final Context context;
    private Map errorsMap;

    @Inject
    public Toaster(Context ctx) {
        context = ctx;
        initialiseErrorStringsMap();
    }

    private void initialiseErrorStringsMap() {
        errorsMap = Maps.newHashMap();
        errorsMap.put(ApiErrorReason.NO_NETWORK_CONNECTION_AVAILABLE, R.string.network_not_connected);
        errorsMap.put(ApiErrorReason.NETWORK_COMMUNICATION_ERROR, R.string.network_communication_failure);
        errorsMap.put(ApiErrorReason.SESSION_DOES_NOT_EXIST, R.string.no_session_to_join);
        errorsMap.put(ApiErrorReason.INVALID_POST_ATTEMPT, R.string.sharing_post_failure);
        errorsMap.put(ApiErrorReason.INVALID_RESPONSE_ATTEMPT, R.string.submit_response_failure);
        errorsMap.put(ApiErrorReason.SESSION_JOIN_FAILED, R.string.session_join_failure);
        errorsMap.put(ApiErrorReason.SESSION_END_FAILED, R.string.end_session_failure);
        errorsMap.put(ApiErrorReason.START_SESSION_FAILED, R.string.session_start_failure);
        errorsMap.put(ApiErrorReason.RAISE_LOWER_HAND_FAILED, R.string.internal_error);
        errorsMap.put(ApiErrorReason.INTERNAL_ERROR, R.string.internal_error);
        errorsMap.put(ApiErrorReason.NO_GOOGLE_CREDENTIALS, R.string.google_authentication_failure);
        errorsMap.put(ApiErrorReason.OPEN_PEAK_ASSOCIATION_FAILED, R.string.open_peak_registration_failed);
        errorsMap.put(ApiErrorReason.PUSH_REGISTRATION_FAILED, R.string.push_notification_registration_failed);

    }


    public void makeToast(int messageId) {
        Toast.makeText(context, context.getString(messageId), Toast.LENGTH_SHORT).show();
    }

    public void makeToast(String message) {
        Toast.makeText(context, message, Toast.LENGTH_SHORT).show();
    }

    public void errorToast(ApiErrorReason reason, int defaultMessage) {
        makeToast(errorsMap.containsKey(reason) ? errorsMap.get(reason) : defaultMessage);
    }

    public void errorToast(ApiErrorReason reason) {
        makeToast(errorsMap.get(reason));
    }

}
