package com.alpha.activities;

import android.content.Intent;
import android.os.Bundle;
import com.alpha.api.services.UserService;
import com.google.inject.Inject;

public class UserInformationActivity extends SundaeTimedActivity {
    public static final String UNREGISTERED = "unregistered";
    public static final String USER_NAME_KEY = "user_name";

    @Inject
    private UserService userService;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);

        Intent intent = new Intent();
        intent.putExtra(USER_NAME_KEY, currentRegisteredUserName());
        setResult(RESULT_OK, intent);
        finish();
    }

    private String currentRegisteredUserName() {
        if (userService.hasUserRegistered()) {
            return userService.getUser().getName();
        } else {
            return UNREGISTERED;
        }
    }
}
