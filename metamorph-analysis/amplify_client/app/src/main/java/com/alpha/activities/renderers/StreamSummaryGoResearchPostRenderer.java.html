package com.alpha.activities.renderers;

import android.content.Context;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ImageView;
import android.widget.TextView;
import com.alpha.api.domains.Post;
import com.alpha.api.services.UserService;
import com.amplify.R;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;

import static java.lang.String.format;

public class StreamSummaryGoResearchPostRenderer {

    private final Post post;
    private final UserService userService;

    @Inject
    public StreamSummaryGoResearchPostRenderer(UserService userService, @Assisted Post post) {
        this.userService = userService;
        this.post = post;
    }

    public void renderView(ViewGroup parentView) {
        TextView title = (TextView) parentView.findViewById(R.id.notebook_post_title);
        ImageView postIcon = (ImageView) parentView.findViewById(R.id.notebook_post_icon);
        Context context = parentView.getContext();

        postIcon.setImageDrawable(context.getResources().getDrawable(R.drawable.icn_stream_goresearch));

        View postContent = parentView.findViewById(R.id.notebook_post_content);
        postContent.setVisibility(View.GONE);

        if(userService.isTeacher()) {
            title.setText(context.getString(R.string.go_research_issued_by_you));
        } else {
            title.setText(format(context.getString(R.string.go_research_issued_by_teacher), post.getAuthor().getName()));
        }
    }
}
