package com.alpha.widgets.text;

import android.content.Context;
import android.content.res.TypedArray;
import android.graphics.Typeface;
import android.util.AttributeSet;
import android.widget.Switch;
import com.amplify.R;
import com.google.common.base.Optional;

import static com.alpha.widgets.text.AmplifyTypeface.DEFAULT_FONT;
import static com.alpha.widgets.text.AmplifyTypeface.Font;
import static com.alpha.widgets.text.AmplifyTypeface.createTypeface;
import static com.alpha.widgets.text.AmplifyTypeface.getFont;

public class AmplifySwitch extends Switch {

    private Optional font = DEFAULT_FONT;

    public AmplifySwitch(Context context) {
        super(context);
    }

    public AmplifySwitch(Context context, AttributeSet attrs) {
        super(context, attrs);
        init(context, attrs);
    }

    public AmplifySwitch(Context context, AttributeSet attrs, int defStyle) {
        super(context, attrs, defStyle);
        init(context, attrs);
    }

    @Override
    public void setSwitchTypeface(Typeface tf, int style) {
        super.setSwitchTypeface(createTypeface(this, font), 0);
    }

    private void init(Context context, AttributeSet attrs) {
        TypedArray config = context.obtainStyledAttributes(attrs, R.styleable.AmplifySwitch);
        font = getFont(config, R.styleable.AmplifySwitch_font);
        config.recycle();
    }
}
