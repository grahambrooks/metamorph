package com.alpha.widgets.text;

import android.content.Context;
import android.content.res.TypedArray;
import android.util.AttributeSet;
import android.widget.EditText;
import com.amplify.R;

import static com.alpha.widgets.text.AmplifyTypeface.applyTypeface;
import static com.alpha.widgets.text.AmplifyTypeface.getFont;

public class AmplifyEditText extends EditText {

    public AmplifyEditText(Context context) {
        super(context);
        applyTypeface(this);
    }

    public AmplifyEditText(Context context, AttributeSet attrs) {
        super(context, attrs);
        init(context, attrs);
    }

    public AmplifyEditText(Context context, AttributeSet attrs, int defStyle) {
        super(context, attrs, defStyle);
        init(context, attrs);
    }

    private void init(Context context, AttributeSet attrs) {
        TypedArray config = context.obtainStyledAttributes(attrs, R.styleable.AmplifyEditText);
        applyTypeface(this, getFont(config, R.styleable.AmplifyEditText_font));
        config.recycle();
    }
}
