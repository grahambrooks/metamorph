package com.alpha.activities;

import android.app.LoaderManager;
import android.content.Context;
import android.content.Loader;
import android.database.Cursor;
import android.os.Bundle;
import com.alpha.activities.adapters.QuickPollResponseCursorAdapter;
import com.alpha.api.adapters.ResponsesCursorLoader;
import com.google.inject.Inject;
import com.google.inject.assistedinject.Assisted;

import static com.alpha.api.repository.sqlite.TableColumn.quick_poll_answer;

public class QuickPollResponseLoaderInitializer implements LoaderManager.LoaderCallbacks {

    private static final Integer QUICK_POLL_RESPONSE_LOADER_ID = 42;
    private final Context context;
    private final Long postId;
    private final QuickPollResponseCursorAdapter quickPollResponseCursorAdapter;

    @Inject
    public QuickPollResponseLoaderInitializer(Context context,
                                              @Assisted Long postId,
                                              @Assisted QuickPollResponseCursorAdapter quickPollResponseCursorAdapter) {
        this.context = context;
        this.postId = postId;
        this.quickPollResponseCursorAdapter = quickPollResponseCursorAdapter;
    }

    @Override
    public Loader onCreateLoader(int id, Bundle args) {
        return new ResponsesCursorLoader(context, postId, quick_poll_answer);
    }

    @Override
    public void onLoadFinished(Loader loader, Cursor cursor) {
        quickPollResponseCursorAdapter.swapCursor(cursor);
    }

    @Override
    public void onLoaderReset(Loader loader) {
        quickPollResponseCursorAdapter.swapCursor(null);
    }

    public void register(LoaderManager loaderManager) {
        loaderManager.initLoader(QUICK_POLL_RESPONSE_LOADER_ID, null, this);
    }
}
