package com.alpha.activities;

import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ListView;
import com.alpha.activities.adapters.CoursePlaylistsCursorAdapter;
import com.amplify.R;
import com.amplify.R.id;
import roboguice.inject.InjectView;

public class MyPlaylistsFragment extends SundaeTimedFragment {

    @InjectView(id.playlists_by_courses)
    private ListView playlists;

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
        return inflater.inflate(R.layout.playlists_layout, container, false);
    }

    @Override
    public void onViewCreated(View view, Bundle savedInstanceState) {
        super.onViewCreated(view, savedInstanceState);

        CoursePlaylistsCursorAdapter cursorAdapter = new CoursePlaylistsCursorAdapter(this.getActivity());
        playlists.setAdapter(cursorAdapter);
    }

}
