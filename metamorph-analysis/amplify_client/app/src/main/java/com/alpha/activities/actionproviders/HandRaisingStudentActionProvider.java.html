public class HandRaisingStudentActionProvider extends SundaeActionProvider {
    private static final Logger LOG = LogUtils.getLogger();

    @Inject
    private HandRaisingListener handRaisingListener;
    @Inject
    private LayoutInflater inflater;
    @Inject
    private EventBus eventBus;

    public HandRaisingStudentActionProvider(Context context) {
        super(context);
        LOG.info("Initing HandRaising Action Provider");
        eventBus.register(this);
    }

    @Override
    public void onResume() {
        LOG.info("Resuming HandRaising Action Provider");
        eventBus.register(this);
        handRaisingListener.updateVisibilityState();
    }

    @Override
    public void onPause() {
        LOG.info("Pausing HandRaising Action Provider");
        eventBus.unregister(this);
    }

    @Override
    public View onCreateActionView() {
        View view = inflater.inflate(R.layout.student_hand_raising_menu, null);
        handRaisingListener.setHandView(view.findViewById(R.id.hand_raised_switch));
        handRaisingListener.updateVisibilityState();
        return view;
    }
