package com.alpha.activities;

import android.content.Context;
import android.database.Cursor;
import com.alpha.api.repository.sqlite.Order;
import com.alpha.api.repository.sqlite.Table;
import com.alpha.api.repository.sqlite.TableColumn;

import static com.alpha.api.repository.sqlite.TableColumn.*;

public class PostResponseRecordsCursorLoader extends BaseCursorLoader {
    private static final String[] PROJECTION = new String[]{_id.name(), author_name.name(), created_on.name(), author_id.name()};
    private static final String SELECTION = TableColumn.post_id.equalTo();
    private static final String SORT_ORDER = Order.ASC.by(created_on);

    public PostResponseRecordsCursorLoader(Context context, PostCursorLoaderCallbacks callbacks, Long postId) {
        super(context, Table.POST_RESPONSE_RECORDS.uri(postId), PROJECTION, SELECTION,  new String[]{String.valueOf(postId)}, SORT_ORDER, callbacks, postId);
    }

    @Override
    public void onLoadFinished(Cursor cursor) {
        callbacks.setResponsesCursor(cursor);
        callbacks.updateResponseCount();
    }

    @Override
    public void onLoaderReset() {
        callbacks.setResponsesCursor(null);
    }

}
