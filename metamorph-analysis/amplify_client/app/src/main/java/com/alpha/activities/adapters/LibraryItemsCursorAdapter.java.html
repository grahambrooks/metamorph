public class LibraryItemsCursorAdapter extends SimpleCursorAdapter{

    private static final String[] FROM = asStrings(TableColumn.title);
    private static final int[] TO = new int[]{R.id.my_library_item_title};

    public LibraryItemsCursorAdapter(Context context) {
        super(context, R.layout.my_library_items_layout,  getCursor(context.getContentResolver()), FROM, TO, CursorAdapter.FLAG_REGISTER_CONTENT_OBSERVER);
    }

    private static Cursor getCursor(ContentResolver contentResolver) {
        return contentResolver.query(Table.LINK_LESSON_COMPONENTS.uri(), asStrings(_id, url, title, image_url), "", new String[]{}, Order.ASC.by(_id));
    }

    @Override
    public void bindView(View view, Context context, Cursor cursor) {
        super.bindView(view, context, cursor);
        SmartImageView imageView = (SmartImageView)view.findViewById(R.id.my_library_item_thumbnail);
        String thumbnailUrl = retrieveStringValue(cursor, TableColumn.image_url);
        imageView.setImageUrl(thumbnailUrl, R.drawable.webpage_icon);
    }
