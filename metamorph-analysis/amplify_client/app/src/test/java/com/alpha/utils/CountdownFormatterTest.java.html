package com.alpha.utils;

import com.xtremelabs.robolectric.RobolectricTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static com.alpha.utils.CountdownFormatter.formatRemainingTimeForStudent;
import static com.alpha.utils.CountdownFormatter.formatRemainingTimeForTeacher;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;

@RunWith(RobolectricTestRunner.class)
@SuppressWarnings("PMD.AvoidUsingHardCodedIP")
public class CountdownFormatterTest {

    @Test
    public void shouldConvertMillisecondsToMinutesAndSecondsForTeacher() {
        assertThat(formatRemainingTimeForTeacher(50000l), is("00:50"));
    }

    @Test
    public void shouldConvertMillisecondsToSecondsWithTwoDigitsForTeacher() {
        assertThat(formatRemainingTimeForTeacher(9000l), is("00:09"));
    }

    @Test
    public void shouldConvertMillisecondsToMinutesWithTwoDigitsForTeacher() {
        assertThat(formatRemainingTimeForTeacher(110000l), is("01:50"));
    }

    @Test
    public void shouldReturnMinuteWhenRemainingTimeIsAMinuteForStudent() {
        assertThat(formatRemainingTimeForStudent(110000l), is("1 min"));
    }

    @Test
    public void shouldReturnMinutesWhenRemainingTimeIsMoreThanOneMinuteForStudent() {
        assertThat(formatRemainingTimeForStudent(130000l), is("2 mins"));
    }

    @Test
    public void shouldReturnSecondsWhenRemainingTimeIsLessThanAMinuteForStudent() {
        assertThat(formatRemainingTimeForStudent(20000l), is("20 sec"));
    }


}
