package com.alpha.activities;

import android.view.LayoutInflater;
import android.view.View;
import android.widget.ListView;
import com.alpha.activities.adapters.CoursePlaylistsCursorAdapter;
import com.alpha.runner.ActivityTestInjector;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.amplify.R.id;
import com.xtremelabs.robolectric.Robolectric;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static com.alpha.runner.ActivityTestInjector.newInjector;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class MyPlaylistsFragmentTest {

    private MyPlaylistsFragment fragment;
    @Mock
    private CoursePlaylistsCursorAdapter adapter;

    @Before
    public void setUp() throws Exception {
        initMocks(this);

        ActivityTestInjector testInjector = newInjector(this, MyPlaylistsFragment.class).bindInstance(CoursePlaylistsCursorAdapter.class, adapter);

        fragment = testInjector.configureFragment();
    }

    @Test
    public void shouldSetAdapterOnViewCreated() {
        View view = fragment.onCreateView(LayoutInflater.from(Robolectric.application), null, null);

        fragment.onViewCreated(view, null);

        ListView listView = (ListView) fragment.getView().findViewById(id.playlists_by_courses);
        assertThat(listView.getAdapter(), is(CoursePlaylistsCursorAdapter.class));

    }
}
