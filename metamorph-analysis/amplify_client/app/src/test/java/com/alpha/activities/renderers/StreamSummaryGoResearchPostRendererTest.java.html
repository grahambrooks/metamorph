package com.alpha.activities.renderers;

import android.view.LayoutInflater;
import android.view.ViewGroup;
import android.widget.TextView;
import com.alpha.api.domains.Post;
import com.alpha.api.domains.PostAuthor;
import com.alpha.api.domains.PostType;
import com.alpha.api.services.UserService;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.amplify.R;
import com.xtremelabs.robolectric.Robolectric;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.hamcrest.core.Is.is;
import static org.junit.Assert.assertThat;
import static org.mockito.BDDMockito.given;
import static org.mockito.Mockito.when;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class StreamSummaryGoResearchPostRendererTest {
    public static final String TEACHER = "Teacher";

    @Mock
    private UserService userService;
    @Mock
    private Post post;
    @Mock
    private PostAuthor postAuthor;

    private StreamSummaryGoResearchPostRenderer renderer;
    private ViewGroup parentView;

    @Before
    public void setUp() {
        initMocks(this);
        renderer = new StreamSummaryGoResearchPostRenderer(userService, post);
        parentView = (ViewGroup) LayoutInflater.from(Robolectric.application).inflate(R.layout.learning_activity_stream_sectioned, null);
        given(postAuthor.getName()).willReturn(TEACHER);
        given(post.getPostType()).willReturn(PostType.GO_RESEARCH);
        given(post.getAuthor()).willReturn(postAuthor);
    }

    @Test
    public void shouldRenderMessageForTeacher() {
        when(userService.isTeacher()).thenReturn(true);

        renderer.renderView(parentView);

        TextView titleView = (TextView) parentView.findViewById(R.id.notebook_post_title);
        assertThat(titleView.getText().toString(), is("You issued a Go Research"));
    }

    @Test
    public void shouldRenderMessageForStudent() {
        when(userService.isTeacher()).thenReturn(false);

        renderer.renderView(parentView);

        TextView titleView = (TextView) parentView.findViewById(R.id.notebook_post_title);
        assertThat(titleView.getText().toString(), is("Teacher issued a Go Research"));
    }
}
