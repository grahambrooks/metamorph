package com.alpha.activities.renderers;

import com.alpha.activities.factories.RendererFactory;
import com.alpha.api.domains.Post;
import com.alpha.api.domains.VideoLearningElement;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.alpha.widget.image.ShadowSmartImageView;
import com.alpha.widgets.image.SmartImageView;
import com.amplify.R;
import com.xtremelabs.robolectric.Robolectric;
import com.xtremelabs.robolectric.shadows.ShadowView;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

import static com.xtremelabs.robolectric.Robolectric.shadowOf;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;

@RunWith(SundaeRobolectricTestRunner.class)
public class SummaryLearningElementRendererVideoTest {
    private static final String PATH_TO_MEDIA = "ABCDEF123";
    private static final String TITLE_TEXT = "Some Text";
    private static final String DESCRIPTION = "description";

    @Mock
    private RendererFactory rendererFactory;
    @Mock
    private Post post;

    private StreamSummarySingleLearningElementRenderer rendererSingle;
    private VideoLearningElement videoElement;

    @Before
    public void setUp() throws Exception {
        MockitoAnnotations.initMocks(this);

        videoElement = new VideoLearningElement(TITLE_TEXT, DESCRIPTION, PATH_TO_MEDIA);
        rendererSingle = new StreamSummarySingleLearningElementRenderer(Robolectric.application, post, rendererFactory);
    }

    @Test
    public void shouldRenderDefaultThumbnailForVideoLearningElementIfCannotGenerateFromMediaPath() {
        ShadowView shadowView = shadowOf(rendererSingle.visit(videoElement, null));

        SmartImageView thumbnail = (SmartImageView) shadowView.findViewById(R.id.video_thumbnail_image);
        ShadowSmartImageView shadowThumbnail = Robolectric.shadowOf_(thumbnail);
        assertThat(shadowThumbnail.getFallbackResource(), is(R.drawable.video_thumbnail_default));
    }
}
