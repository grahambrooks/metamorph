package com.alpha.widgets;

import android.view.GestureDetector;
import android.view.MotionEvent;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.xtremelabs.robolectric.Robolectric;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.Matchers.any;
import static org.mockito.Mockito.verify;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class GestureViewFlipperTest {

    private GestureViewFlipper viewFlipper;

    @Mock
    private GestureDetector gestureDetector;

    @Before
    public void setUp() throws Exception {
        initMocks(this);

        viewFlipper = new GestureViewFlipper(Robolectric.application, gestureDetector);
    }

    @Test
    public void shouldInvokeOnGestureListenerOnInterceptTouchEvent() {
        viewFlipper.onInterceptTouchEvent(null);

        verify(gestureDetector).onTouchEvent(any(MotionEvent.class));
    }
}
