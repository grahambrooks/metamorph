package com.alpha.runner.shadows;

import android.content.BroadcastReceiver;
import android.content.BroadcastReceiver.PendingResult;
import com.xtremelabs.robolectric.internal.Implementation;
import com.xtremelabs.robolectric.internal.Implements;

@Implements(BroadcastReceiver.class)
public class ShadowBroadcastReceiver {
    private PendingResult pendingResult;
    private boolean hasGoneAsync = false;

    public boolean hasGoneAsync() {
        return hasGoneAsync;
    }

    public PendingResult getPendingResult() {
        return pendingResult;
    }

    public void setPendingResult(PendingResult pendingResult) {
        this.pendingResult = pendingResult;
    }

    @Implementation
    public final PendingResult goAsync() {
        hasGoneAsync = true;
        return pendingResult;
    }
}
