package com.alpha.activities;

import android.content.CursorLoader;
import android.database.Cursor;
import com.alpha.activities.adapters.QuickPollResponseCursorAdapter;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.xtremelabs.robolectric.Robolectric;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;
import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.verify;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class QuickPollResponseLoaderInitializerTest {
    @Mock
    private QuickPollResponseCursorAdapter quickPollResponseCursorAdapter;

    private QuickPollResponseLoaderInitializer loaderInitializer;
    private static final Long POST_ID = 1l;

    @Before
    public void setUp() {
        initMocks(this);
        loaderInitializer = new QuickPollResponseLoaderInitializer(Robolectric.application, POST_ID, quickPollResponseCursorAdapter);
    }

    @Test
    public void shouldReturnCursorLoaderWithCommentsURI() throws Exception {
        CursorLoader cursorLoader = (CursorLoader) loaderInitializer.onCreateLoader(PostCursorLoaderInitializer.COMMENTS_CURSOR_LOADER_ID, null);
        assertThat(cursorLoader.getUri().toString(), is("content://com.alpha.api.Sundae/posts/" +
                POST_ID + "/responses"));
    }

    @Test
    public void shouldSwapCursoronLoadFinished() {
        Cursor cursor = mock(Cursor.class);

        loaderInitializer.onLoadFinished(null, cursor);

        verify(quickPollResponseCursorAdapter).swapCursor(cursor);
    }
}
