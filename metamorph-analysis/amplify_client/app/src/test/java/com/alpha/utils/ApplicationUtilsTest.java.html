package com.alpha.utils;

import android.content.Intent;
import android.content.pm.ActivityInfo;
import android.content.pm.ResolveInfo;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.xtremelabs.robolectric.Robolectric;
import com.xtremelabs.robolectric.res.RobolectricPackageManager;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import java.util.List;

import static com.google.common.collect.Lists.newArrayList;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;
import static org.mockito.Mockito.mock;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class ApplicationUtilsTest {
    private ApplicationUtils applicationUtils;
    private RobolectricPackageManager roboPackageManager;
    private Intent setupIntent;

    private ResolveInfo resolveInfo;
    private ActivityInfo activityInfo;

    @Before
    public void setup() {
        initMocks(Robolectric.application);
        roboPackageManager = (RobolectricPackageManager) Robolectric.application.getPackageManager();

        applicationUtils = new ApplicationUtils();

        setupIntent = new Intent(Intent.ACTION_MAIN);
        setupIntent.addCategory(Intent.CATEGORY_LAUNCHER);

        resolveInfo = mock(ResolveInfo.class);
        activityInfo = mock(ActivityInfo.class);
    }

    @Test
    public void shouldReturnTrueIfAppExists() {
        activityInfo.packageName = "com.sample.app";
        resolveInfo.activityInfo = activityInfo;

        roboPackageManager.addResolveInfoForIntent(setupIntent, newArrayList(resolveInfo));

        assertThat(applicationUtils.isAppInstalled(Robolectric.application, "com.sample.app"), is(true));
    }

    @Test
    public void shouldNotReturnEmptyListOfApps() {
        roboPackageManager.addResolveInfoForIntent(setupIntent, newArrayList(resolveInfo, resolveInfo));

        List listOfApps = applicationUtils.getListOfApps(Robolectric.application);
        assertThat(listOfApps.size(), is(2));
    }


}
