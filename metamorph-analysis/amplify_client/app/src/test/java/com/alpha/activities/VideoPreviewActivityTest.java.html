package com.alpha.activities;

import android.content.Intent;
import android.os.Bundle;
import android.webkit.WebView;
import com.alpha.runner.ActivityTestInjector;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.google.inject.Inject;
import com.xtremelabs.robolectric.shadows.ShadowActivity;
import com.xtremelabs.robolectric.shadows.ShadowWebView;
import org.junit.Ignore;
import org.junit.Test;
import org.junit.runner.RunWith;

import static com.xtremelabs.robolectric.Robolectric.shadowOf;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;
import static org.mockito.Mockito.*;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class VideoPreviewActivityTest {

    @Inject
    private VideoPreviewActivity videoPreviewActivity;

    private final ActivityTestInjector activityTestInjector = new ActivityTestInjector(this, VideoPreviewActivity.class);

    @Test
    @Ignore
    public void testShouldDisplayVideo() throws Exception {
        // TODO: I think we should ask Cassie for Help on this. Also, I need to read up on
        // using Robolectric to test activities.
        initMocks(this);

        activityTestInjector.configureActivity();

        ShadowActivity shadow = shadowOf(videoPreviewActivity);
        videoPreviewActivity.onCreate(null);
        videoPreviewActivity.onResume();

        ShadowWebView shadowWebView = shadowOf((WebView) shadow.getContentView());

        assertThat(shadowWebView.getLastLoadData().data.contains("youtube"), is(true));
    }

    @Test
    public void activityShouldReadElementFromBundle() throws Exception {
        initMocks(this);

        activityTestInjector.configureActivity();

        Intent intent = mock(Intent.class);
        Bundle bundle = mock(Bundle.class);
        when(intent.getExtras()).thenReturn(bundle);
        videoPreviewActivity.setIntent(intent);

        videoPreviewActivity.onCreate(null);

        verify(bundle).get(VideoPreviewActivity.VIDEO_LEARNING_ACTIVITY);
    }
}
