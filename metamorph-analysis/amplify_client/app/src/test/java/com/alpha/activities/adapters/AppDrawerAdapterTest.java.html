package com.alpha.activities.adapters;

import android.content.Intent;
import android.graphics.drawable.Drawable;
import android.view.View;
import android.widget.TextView;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.amplify.R;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import static com.xtremelabs.robolectric.Robolectric.shadowOf;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;

@RunWith(SundaeRobolectricTestRunner.class)
public class AppDrawerAdapterTest extends AppInfoAdapterTestBase {

    @Before
    public void setUp() {
        setUpMocks();
        appDrawerAdapter = new AppDrawerAdapter(context);
    }

    @Test
    public void shouldLaunchApplicationWhenIconClicked() {
        appDrawerAdapter.add(sundaeAppInfo);
        TextView view = (TextView) appDrawerAdapter.getView(0, null, null).findViewById(R.id.app_icon_text_view);
        view.performClick();

        Intent intent = shadowOf(context).getNextStartedActivity();
        assertThat(intent.getComponent().getPackageName(), is("package"));
    }

    @Test
    public void shouldLoadAppInfoIntoView() {
        appDrawerAdapter.add(sundaeAppInfo);
        View appLayout = appDrawerAdapter.getView(0, null, null);
        TextView appIcon = (TextView)appLayout.findViewById(getTextViewId());
        assertThat(appIcon.getText().toString(), is("My eReader"));

        Drawable drawable = appIcon.getCompoundDrawables()[1];
        assertThat(drawable, is(icon));
    }

    @Override
    protected int getTextViewId() {
        return R.id.app_icon_text_view;
    }

    @Override
    protected int getAppInfoViewId() {
        return R.layout.app_drawer_icon;
    }
}
