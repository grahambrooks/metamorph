package com.alpha.activities.adapters;

import com.alpha.activities.mappings.NotebooksCoverMappings;
import com.alpha.activities.view.AmplifyCheckableView;
import com.alpha.api.services.Section;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.xtremelabs.robolectric.Robolectric;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import java.util.List;

import static com.google.common.collect.Lists.newArrayList;
import static org.junit.Assert.assertTrue;
import static org.mockito.Mockito.mock;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class NotebookPreviewAdapterTest {
    private List
 sections;

    @Mock
    Section sectionOne;


    @Before
    public void setUp() throws Exception {
        initMocks(this);
        sections = newArrayList(sectionOne, mock(Section.class));
    }

    @Test
    public void shouldHaveAllSectionsCheckedByDefault() throws Exception {
        NotebookPreviewAdapter adapter = new NotebookPreviewAdapter(sections, new NotebooksCoverMappings(), Robolectric.application);

        assertTrue(adapter.getCount() == 2);
        assertTrue("All items should be selected by default", adapter.getSelectedSectionsCount() == 2);
    }

    @Test
    public void shouldTrackCheckedSections() throws Exception {
        NotebookPreviewAdapter adapter = new NotebookPreviewAdapter(sections, mock(NotebooksCoverMappings.class), Robolectric.application);


        AmplifyCheckableView firstItemOnGrid = (AmplifyCheckableView) adapter.getView(0, null, null);
        firstItemOnGrid.toggle();

        assertTrue(adapter.getCount() == 2);
        assertTrue("Only one item should be checked", adapter.getSelectedSectionsCount() == 1);

    }
}
