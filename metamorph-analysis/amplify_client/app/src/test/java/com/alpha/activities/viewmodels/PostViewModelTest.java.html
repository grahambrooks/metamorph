package com.alpha.activities.viewmodels;

import org.apache.commons.lang3.StringUtils;
import org.junit.Before;
import org.junit.Test;

import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;

public class PostViewModelTest {

    private PostViewModel viewModel;

    @Before
    public void setUp() throws Exception {
        viewModel = new PostViewModel();
    }

    @Test
    public void shouldBeValidIfTextIsNotBlank() {
        viewModel.setText("something");

        assertThat(viewModel.isValid(), is(true));
    }

    @Test
    public void testTrimText() throws Exception {
        viewModel.setText("something    ");

        assertThat(viewModel.getModel(), is("something"));
    }

    @Test
    public void testWorkForNullText() throws Exception {
        viewModel.setText(null);

        assertThat(viewModel.getModel(), is(StringUtils.EMPTY));
    }

    @Test
    public void shouldNotBeValidIfTextIsBlank() {
        viewModel.setText(StringUtils.EMPTY);

        assertThat(viewModel.isValid(), is(false));
    }
}
