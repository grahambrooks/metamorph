package com.alpha.activities;

import android.app.Activity;
import com.alpha.logging.AnalyticsAgent;
import com.alpha.runner.ActivityTestInjector;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.google.inject.Inject;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.mockito.Matchers.eq;
import static org.mockito.Mockito.verify;

@RunWith(SundaeRobolectricTestRunner.class)
public class SundaeTimedFragmentTest {
    @Inject
    private AnalyticsAgent analyticsAgent;
    @Inject
    private Activity context;

    public static class TestTimedFragment extends SundaeTimedFragment {};

    private final ActivityTestInjector activityTestInjector =
            new ActivityTestInjector(this, TestTimedFragment.class);

    private SundaeTimedFragment fragment;

    @Before
    public void setUp() throws Exception {
        fragment = activityTestInjector.configureFragment();
    }

    @Test
    public void shouldStartAnalyticsSessionOnStart() {
        // onStart is called by the configureFragment retrieve in setUp

        verify(analyticsAgent).startSession(context);
    }

    @Test
    public void shouldStartTimedEventOnResume() {
        fragment.onResume();

        verify(analyticsAgent).startTimedEvent(eq(TestTimedFragment.class.getSimpleName()));
    }

    @Test
    public void shouldStopTimedEventOnPause() {
        fragment.onPause();

        verify(analyticsAgent).endTimedEvent(eq(TestTimedFragment.class.getSimpleName()));
    }

    @Test
    public void shouldStopAnalyticsSessionOnStop() {
        fragment.onStop();

        verify(analyticsAgent).stopSession(context);
    }
}
