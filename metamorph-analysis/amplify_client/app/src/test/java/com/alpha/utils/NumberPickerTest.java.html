package com.alpha.utils;

import org.junit.Test;

import static com.alpha.utils.NumberPicker.DEFAULT_TIMER_VALUE;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;

public class NumberPickerTest {
    private NumberPicker picker;

    @Test
    public void shouldIncrementCountByOneForNextValue() {
        picker = new NumberPicker(1, 20);

        assertThat(picker.increment(), is(DEFAULT_TIMER_VALUE + 1));
    }

    @Test
    public void shouldDecrementCountByOneForPreviousValue() {
        picker = new NumberPicker(2, 20);

        picker.increment();
        assertThat(picker.decrement(), is(DEFAULT_TIMER_VALUE));
    }

    @Test
    public void shouldNotIncrementMoreThanMaxValue() {
        picker = new NumberPicker(4, 20);

        assertThat(picker.increment(), is(DEFAULT_TIMER_VALUE + 1));
        assertThat(picker.increment(), is(DEFAULT_TIMER_VALUE + 2));
    }

    @Test
    public void shouldNotDecrementLessThanStartValue() {
        picker = new NumberPicker(4, 20);

        assertThat(picker.decrement(), is(DEFAULT_TIMER_VALUE - 1));
    }

    @Test
    public void shouldUseTheCurrentValueForCalculatingNextValue() {
        picker = new NumberPicker(4, 20);
        picker.setCurrent(5);

        assertThat(picker.increment(), is(6));
    }

    @Test
    public void shouldNotUpdateCurrentValueIfItIsGreaterThanMaxValue() {
        picker = new NumberPicker(4, 20);
        picker.setCurrent(7);

        assertThat(picker.increment(), is(8));
    }

    @Test
    public void shouldNotUpdateCurrentValueIfItIsLesserThanStartValue() {
        picker = new NumberPicker(4, 20);
        picker.setCurrent(3);

        assertThat(picker.increment(), is(DEFAULT_TIMER_VALUE + 1));
    }
}
