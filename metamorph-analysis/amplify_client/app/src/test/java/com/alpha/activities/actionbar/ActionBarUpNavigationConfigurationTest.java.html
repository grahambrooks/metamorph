@RunWith(SundaeRobolectricTestRunner.class)
public class ActionBarUpNavigationConfigurationTest {
    @Inject
    private ActionBarUpNavigationConfiguration actionBarConfiguration;
    private Activity activity;

    @Before
    public void setUp() throws Exception {
        ActivityTestInjector testInjector =
                new ActivityTestInjector(this, ActionBarUpNavigationConfiguration.class);
        testInjector.configureTestSubject();
        activity = testInjector.getCurrentActivityContext();
        invokeOnCreateAndSetContentView();
    }

    private void invokeOnCreateAndSetContentView() throws NoSuchMethodException, IllegalAccessException, InvocationTargetException {
        Method onCreate = Activity.class.getDeclaredMethod("onCreate", Bundle.class);
        onCreate.setAccessible(true);
        onCreate.invoke(activity, new Object[]{null});
        activity.setContentView(new View(Robolectric.application));
    }

    @Test
    public void shouldEnableHomeAsUpNavigation() {
        actionBarConfiguration.setIconForUpNavigation(0);

        assertThat(getActionBar().getDisplayOptions() & ActionBar.DISPLAY_HOME_AS_UP, is(ActionBar.DISPLAY_HOME_AS_UP));
    }

    @Test
    public void shouldEnableDisplayShowHome() {
        actionBarConfiguration.setIconForUpNavigation(0);

        assertThat(getActionBar().getDisplayOptions() & ActionBar.DISPLAY_SHOW_HOME, is(ActionBar.DISPLAY_SHOW_HOME));
    }

    @Test
    public void shouldDisableDisplayShowTitle() {
        actionBarConfiguration.setIconForUpNavigation(0);

        assertThat(getActionBar().getDisplayOptions() & ActionBar.DISPLAY_SHOW_TITLE, not(ActionBar.DISPLAY_SHOW_TITLE));
    }
