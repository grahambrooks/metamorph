package com.alpha.utils;

import com.alpha.api.services.events.TimeTickerEndEvent;
import com.alpha.api.services.events.TimeTickerEvent;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.google.common.eventbus.EventBus;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static org.mockito.Matchers.isA;
import static org.mockito.Mockito.verify;
import static org.mockito.MockitoAnnotations.initMocks;

@SuppressWarnings("PMD")
@RunWith(SundaeRobolectricTestRunner.class)
public class TimeTickerTest {

    @Mock
    private EventBus eventBus;
    private TimeTicker timeTicker;

    @Before
    public void setUp() {
        initMocks(this);
        timeTicker = new TimeTicker(30000l, 1000l, eventBus);
    }

    @Test
    public void shouldPostEventOnTick() {
        timeTicker.onTick(1L);
        verify(eventBus).post(isA(TimeTickerEvent.class));
    }

    @Test
    public void shouldPostEventOnFinish() {
        timeTicker.onFinish();
        verify(eventBus).post(isA(TimeTickerEndEvent.class));
    }
}
