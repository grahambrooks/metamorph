package com.alpha.utils;

import android.os.Bundle;
import com.xtremelabs.robolectric.RobolectricTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import java.io.Serializable;

import static org.hamcrest.core.Is.is;
import static org.junit.Assert.assertThat;

@RunWith(RobolectricTestRunner.class)
public class BundlesTest {
    private static class MyObject implements Serializable {}

    @Test
    public void shouldCreateABundleWithItem() throws Exception {
        MyObject item = new MyObject();
        Bundle bundle = Bundles.withItems(item);

        assertThat((MyObject) bundle.getSerializable(MyObject.class.getName()), is(item));
    }

    @Test
    public void shouldRetrieveAnItemFromBundle() throws Exception {
        Bundle bundle = new Bundle();
        MyObject item = new MyObject();
        bundle.putSerializable(MyObject.class.getName(), item);

        assertThat(Bundles.get(bundle, MyObject.class) , is(item));
    }
}
