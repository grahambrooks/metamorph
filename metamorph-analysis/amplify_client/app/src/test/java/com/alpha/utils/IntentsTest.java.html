package com.alpha.utils;

import android.content.Intent;
import android.os.Bundle;
import com.alpha.runner.SundaeRobolectricTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.is;
import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.when;

@RunWith(SundaeRobolectricTestRunner.class)
public class IntentsTest {
    @Test
    public void shouldCheckIfIntentHasNoExtraWhenNull() throws Exception {
        assertThat(Intents.hasNoIntentExtras(null), is(true));
    }

    @Test
    /**
     * Roboelectric fills empty bundle when intent is created,
     * but the app fails in android environment because when no extras are added to the intent, bundle is null.
     * This test explicitly simulates bundle being null.
     */
    public void shouldCheckIfIntentHasNoExtraWhenBundleIsNull() throws Exception {
        Intent intent = mock(Intent.class);
        when(intent.getExtras()).thenReturn(null);

        assertThat(Intents.hasNoIntentExtras(intent), is(true));
    }

    @Test
    public void shouldCheckIfIntentHasNoExtraWhenBundleIsEmpty() throws Exception {
        Intent intent = new Intent();
        intent.putExtras(new Bundle());
        assertThat(Intents.hasNoIntentExtras(intent), is(true));
    }
}
