package com.alpha.activities.mappings;

import com.amplify.R;
import org.junit.Before;
import org.junit.Test;

import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.core.Is.is;

public class NotebooksCoverMappingsTest {

    public static final String SECTION_NAME = "section";

    private NotebooksCoverMappings coverMappings;

    @Before
    public void setUp() throws Exception {
        coverMappings = new NotebooksCoverMappings();
    }

    @Test
    public void shouldGetDefaultCoverImage() {
        assertThat(coverMappings.getHomeCoverImage(SECTION_NAME), is(NotebooksCoverMappings.DEFAULT_COVERS.getHomeCoverImage()));
    }

    @Test
    public void shouldGetDefaultCoverIcon() {
        assertThat(coverMappings.getShareCoverImage(SECTION_NAME), is(NotebooksCoverMappings.DEFAULT_COVERS.getShareCoverImage()));
    }

    @Test
    public void shouldGetDefaultCoverImageByPartialNameMatching() {
        assertThat(coverMappings.getHomeCoverImage("Math Horntail(UMD)"), is(R.drawable.img_home_math));
    }

    @Test
    public void shouldReturnDefaultNotebookCoverIdIfNoSectionNameExists() {
        assertThat(coverMappings.getShareCoverImage("ASdasd"), is(NotebooksCoverMappings.DEFAULT_COVERS.getShareCoverImage()));
    }

    @Test
    public void shouldGetDefaultCoverImageByStripingLastWordOut() {
        assertThat(coverMappings.getHomeCoverImage("writing (AB)"), is(R.drawable.img_home_languagearts));
    }
}
