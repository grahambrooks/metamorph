package com.alpha.utils;

import org.junit.Test;

import static org.hamcrest.core.Is.is;
import static org.junit.Assert.assertThat;

public class GoogleDriveUtilsTest {
    @Test
    public void shouldStripFileIDFromDocumentURL() throws Exception {
        String fileId = "1TdUopScvLVvtF4MvnMo-syaLCAgkbOwhnn8_LXcsvPE";
        String documentURL = "https://docs.google.com/a/amplifydev.net/document/d/" + fileId + "/edit";

        final String actualFileId = new GoogleDriveUtils().getFileId(documentURL);
        assertThat(actualFileId, is(fileId));
    }

    @Test
    public void shouldGetFileIDFromSpreadSheetURL() throws Exception {
        String fileId = "0Au8Qs8F5MYxKdGk0WGdya2VJWHF5bkc1U1JnM0w5eVE";
        String documentURL = "https://docs.google.com/a/amplifydev.net/spreadsheet/ccc?key=" + fileId + "#gid=0";

        final String actualFileId = new GoogleDriveUtils().getFileId(documentURL);
        assertThat(actualFileId, is(fileId));
    }

    @Test
    public void shouldGetFileIDFromSpreadSheetURLWithoutGid() throws Exception {
        String fileId = "0Au8Qs8F5MYxKdGk0WGdya2VJWHF5bkc1U1JnM0w5eVE";

    }
