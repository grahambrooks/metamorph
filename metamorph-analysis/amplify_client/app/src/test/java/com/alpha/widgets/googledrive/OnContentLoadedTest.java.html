    @Test
    public void shouldProcessRegisteredLoadAction() throws Exception {
        final File file = new File();

        OnContentLoaded contentLoaded = new OnContentLoaded(new Callable() {
            @Override
            public File call() throws Exception {
                return file;
            }
        });

        contentLoaded.register(loadAction);

        //hoping the callback will be returned before the timeout reaches.
        verify(loadAction, timeout(5000)).onSuccess(file);
    }

    @Test

                throw expectedError;
            }
