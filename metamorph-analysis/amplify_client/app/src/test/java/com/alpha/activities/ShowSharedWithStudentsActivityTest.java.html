package com.alpha.activities;

import android.content.Intent;
import android.widget.TextView;
import com.alpha.api.services.Participant;
import com.alpha.api.services.Section;
import com.alpha.api.services.UserService;
import com.alpha.runner.SundaeRobolectricTestRunner;
import com.amplify.R;
import com.google.inject.Inject;
import org.junit.Before;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.mockito.Mock;

import static com.alpha.runner.ActivityTestInjector.newInjector;
import static com.google.common.collect.Lists.newArrayList;
import static org.hamcrest.Matchers.is;
import static org.junit.Assert.assertThat;
import static org.mockito.BDDMockito.given;
import static org.mockito.MockitoAnnotations.initMocks;

@RunWith(SundaeRobolectricTestRunner.class)
public class ShowSharedWithStudentsActivityTest {

    private final Participant participant = new Participant(1L, "Sam", true);

    @Inject ShowSharedWithStudentsActivity activity;

    @Mock UserService userService;
    @Mock Section section;

    @Before
    public void setUp() throws Exception {
        initMocks(this);

        newInjector(this, ShowSharedWithStudentsActivity.class)
                .bindInstance(UserService.class, userService)
                .configureActivity();
    }

    @Test
    public void shouldDisplaySharedCount() {
        given(userService.getCurrentUserSection()).willReturn(section);
        given(section.getAllParticipantsCount()).willReturn(3);

        Intent intent = new Intent();
        intent.putExtra(ShowSharedWithStudentsActivity.PARTICIPANTS, newArrayList(participant));
        activity.setIntent(intent);

        activity.onCreate(null);

        TextView view = (TextView) activity.findViewById(R.id.show_shared_with_students_count);
        assertThat(view.getText().toString(), is("1/3 students"));
    }
}
