package com.amplify.environment.receivers;

import android.content.BroadcastReceiver;
import android.content.Context;
import android.content.Intent;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class ReinstallReceiver extends BroadcastReceiver {

    protected static final String RESET_STATE_INTENT = "com.amplify.loaner.RESET_STATE";
    private static final Logger LOG = LoggerFactory.getLogger(ReinstallReceiver.class);

    @Override
    public void onReceive(Context context, Intent intent) {
        LOG.info("Initiating notebooks reset...");

        Intent broadcastIntent = new Intent();
        broadcastIntent.setAction(RESET_STATE_INTENT);
        context.sendOrderedBroadcast(broadcastIntent, null);
    }
}
