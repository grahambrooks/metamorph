package com.amplify.environment.receivers;

import android.content.Context;
import android.content.Intent;
import com.xtremelabs.robolectric.RobolectricTestRunner;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.verify;

@RunWith(RobolectricTestRunner.class)
public class ReinstallReceiverTest {


    @Test
    public void shouldBroadcastNotebooksResetIntent() throws Exception {
        Context context = mock(Context.class);
        ReinstallReceiver receiver = new ReinstallReceiver();
        receiver.onReceive(context, null);

        Intent expectedIntent = new Intent();
        expectedIntent.setAction(ReinstallReceiver.RESET_STATE_INTENT);
        verify(context).sendOrderedBroadcast(expectedIntent, null);
    }
}
